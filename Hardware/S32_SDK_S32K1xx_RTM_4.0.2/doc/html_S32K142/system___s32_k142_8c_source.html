53 /*FUNCTION**********************************************************************
54  *
55  * Function Name : SystemInit
56  * Description : This function disables the watchdog, enables FPU
57  * and the power mode protection if the corresponding feature macro
58  * is enabled. SystemInit is called from startup_device file.
59  *
60  * Implements : SystemInit_Activity
61  *END**************************************************************************/
62 void SystemInit(void)
63 {
64 /**************************************************************************/
65  /* FPU ENABLE*/
66 /**************************************************************************/
67 #ifdef ENABLE_FPU
68  /* Enable CP10 and CP11 coprocessors */
69  S32_SCB->CPACR |= (S32_SCB_CPACR_CP10_MASK | S32_SCB_CPACR_CP11_MASK);
70 #ifdef ERRATA_E6940
71  /* Disable lazy context save of floating point state by clearing LSPEN bit
72  * Workaround for errata e6940 */
73  S32_SCB->FPCCR &= ~(S32_SCB_FPCCR_LSPEN_MASK);
74 #endif
75 #endif /* ENABLE_FPU */
76 
77 /**************************************************************************/
78  /* WDOG DISABLE*/
79 /**************************************************************************/
80 
81 #if (DISABLE_WDOG)
82  /* Write of the WDOG unlock key to CNT register, must be done in order to allow any modifications*/
83  WDOG->CNT = (uint32_t ) FEATURE_WDOG_UNLOCK_VALUE;
84  /* The dummy read is used in order to make sure that the WDOG registers will be configured only
85  * after the write of the unlock value was completed. */
86  (void)WDOG->CNT;
87 
88  /* Initial write of WDOG configuration register:
89  * enables support for 32-bit refresh/unlock command write words,
90  * clock select from LPO, update enable, watchdog disabled */
91  WDOG->CS = (uint32_t ) ( (1UL << WDOG_CS_CMD32EN_SHIFT) |
92  (FEATURE_WDOG_CLK_FROM_LPO << WDOG_CS_CLK_SHIFT) |
93  (0U << WDOG_CS_EN_SHIFT) |
94  (1U << WDOG_CS_UPDATE_SHIFT) );
95 
96  /* Configure timeout */
97  WDOG->TOVAL = (uint32_t )0xFFFF;
98 #endif /* (DISABLE_WDOG) */
99 
100 /**************************************************************************/
101  /* ENABLE CACHE */
102 /**************************************************************************/
103 #if defined(I_CACHE) && (ICACHE_ENABLE == 1)
104  /* Invalidate and enable code cache */
105  LMEM->PCCCR = LMEM_PCCCR_INVW0(1) | LMEM_PCCCR_INVW1(1) | LMEM_PCCCR_GO(1) | LMEM_PCCCR_ENCACHE(1);
106 #endif /* defined(I_CACHE) && (ICACHE_ENABLE == 1) */
107 }

109 /*FUNCTION**********************************************************************
110  *
111  * Function Name : SystemCoreClockUpdate
112  * Description : This function must be called whenever the core clock is changed
113  * during program execution. It evaluates the clock register settings and calculates
114  * the current core clock.
115  *
116  * Implements : SystemCoreClockUpdate_Activity
117  *END**************************************************************************/
118 void SystemCoreClockUpdate(void)
119 {
120  uint32_t SCGOUTClock = 0U; /* Variable to store output clock frequency of the SCG module */
121  uint32_t regValue; /* Temporary variable */
122  uint32_t divider, prediv, multi;
123  bool validSystemClockSource = true;
124  static const uint32_t fircFreq[] = {
125  FEATURE_SCG_FIRC_FREQ0,
126  };
127 
128  divider = ((SCG->CSR & SCG_CSR_DIVCORE_MASK) >> SCG_CSR_DIVCORE_SHIFT) + 1U;
129 
130  switch ((SCG->CSR & SCG_CSR_SCS_MASK) >> SCG_CSR_SCS_SHIFT) {
131  case 0x1:
132  /* System OSC */
133  SCGOUTClock = CPU_XTAL_CLK_HZ;
134  break;
135  case 0x2:
136  /* Slow IRC */
137  regValue = (SCG->SIRCCFG & SCG_SIRCCFG_RANGE_MASK) >> SCG_SIRCCFG_RANGE_SHIFT;
138 
139  if (regValue != 0U)
140  {
141  SCGOUTClock = FEATURE_SCG_SIRC_HIGH_RANGE_FREQ;
142  }
143 
144  break;
145  case 0x3:
146  /* Fast IRC */
147  regValue = (SCG->FIRCCFG & SCG_FIRCCFG_RANGE_MASK) >> SCG_FIRCCFG_RANGE_SHIFT;
148  SCGOUTClock= fircFreq[regValue];
149  break;
150  case 0x6:
151  /* System PLL */
152  SCGOUTClock = CPU_XTAL_CLK_HZ;
153  prediv = ((SCG->SPLLCFG & SCG_SPLLCFG_PREDIV_MASK) >> SCG_SPLLCFG_PREDIV_SHIFT) + 1U;
154  multi = ((SCG->SPLLCFG & SCG_SPLLCFG_MULT_MASK) >> SCG_SPLLCFG_MULT_SHIFT) + 16U;
155  SCGOUTClock = SCGOUTClock * multi / (prediv * 2U);
156  break;
157  default:
158  validSystemClockSource = false;
159  break;
160  }
161 
162  if (validSystemClockSource == true) {
163  SystemCoreClock = (SCGOUTClock / divider);
164  }
165 }

167 /*FUNCTION**********************************************************************
168  *
169  * Function Name : SystemSoftwareReset
170  * Description : This function is used to initiate a system reset
171  *
172  * Implements : SystemSoftwareReset_Activity
173  *END**************************************************************************/
174 void SystemSoftwareReset(void)
175 {
176  uint32_t regValue;
177 
178  /* Read Application Interrupt and Reset Control Register */
179  regValue = S32_SCB->AIRCR;
180 
181  /* Clear register key */
182  regValue &= ~( S32_SCB_AIRCR_VECTKEY_MASK);
183 
184  /* Configure System reset request bit and Register Key */
185  regValue |= S32_SCB_AIRCR_VECTKEY(FEATURE_SCB_VECTKEY);
186  regValue |= S32_SCB_AIRCR_SYSRESETREQ(0x1u);
187 
188  /* Write computed register value */
189  S32_SCB->AIRCR = regValue;
190 }
