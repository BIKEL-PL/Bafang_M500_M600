39 /*FUNCTION**********************************************************************
40  *
41  * Function Name : FTM_DRV_QuadDecodeStart
42  * Description : Configures the parameters needed and activates quadrature
43  * decode mode.
44  *
45  * Implements : FTM_DRV_QuadDecodeStart_Activity
46  *END**************************************************************************/
47 status_t FTM_DRV_QuadDecodeStart(uint32_t instance,
48  const ftm_quad_decode_config_t * config)
49 {
50  DEV_ASSERT((instance == 1U) || (instance == 2U));
51  DEV_ASSERT(config != NULL);
52  FTM_Type * ftmBase = g_ftmBase[instance];
53  ftm_state_t * state = ftmStatePtr[instance];
54  status_t retStatus = STATUS_SUCCESS;
55 
56  if ((NULL != state) && (FTM_MODE_NOT_INITIALIZED == state->ftmMode))
57  {
58  /* Disable Quadrature Decoder */
59  FTM_DRV_SetQuadDecoderCmd(ftmBase, false);
60  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);
61  /* Configure Quadrature Decoder */
62  /* Set decoder mode Speed and direction or Phase A and Phase B encoding */
63  FTM_DRV_SetQuadMode(ftmBase, (uint8_t)config->mode);
64  /* Set filter state for Phase A (enable/disable) */
65  FTM_DRV_SetQuadPhaseAFilterCmd(ftmBase, config->phaseAConfig.phaseInputFilter);
66  /* Set Phase A filter value if phase filter is enabled */
67  if (config->phaseAConfig.phaseInputFilter)
68  {
69  FTM_DRV_SetChnInputCaptureFilter(ftmBase, CHAN0_IDX, config->phaseAConfig.phaseFilterVal);
70  }
71 
72  /* Set filter state for Phase B (enable/disable) */
73  FTM_DRV_SetQuadPhaseBFilterCmd(ftmBase, config->phaseBConfig.phaseInputFilter);
74  /* Set Phase B filter value if phase filter is enabled */
75  if (config->phaseBConfig.phaseInputFilter)
76  {
77  FTM_DRV_SetChnInputCaptureFilter(ftmBase, CHAN1_IDX, config->phaseBConfig.phaseFilterVal);
78  }
79 
80  /* Set polarity for Phase A and Phase B */
81  FTM_DRV_SetQuadPhaseAPolarity(ftmBase, (uint8_t)config->phaseAConfig.phasePolarity);
82  FTM_DRV_SetQuadPhaseBPolarity(ftmBase, (uint8_t)config->phaseBConfig.phasePolarity);
83  /* Configure counter (initial value and maximum value) */
84  FTM_DRV_SetCounterInitVal(ftmBase, config->initialVal);
85  FTM_DRV_SetMod(ftmBase, config->maxVal);
86  FTM_DRV_SetCounter(ftmBase, config->initialVal);
87  /* Enable Quadrature Decoder */
88  FTM_DRV_SetQuadDecoderCmd(ftmBase, true);
89  state->ftmMode = FTM_MODE_QUADRATURE_DECODER;
90  }
91  else
92  {
93  retStatus = STATUS_ERROR;
94  }
95 
96  return retStatus;
97 }
