342  emios_oc_param_t ocParam;
343  emios_bus_select_t timeBaseSelection = EMIOS_BUS_SEL_BCDE;
344  static const IRQn_Type emiosIrq[eMIOS_INSTANCE_COUNT][(eMIOS_UC_COUNT / 2U)] = eMIOS_IRQS;
345 
346  /* Allocate one of the OC state structure for this instance */
347  instancePtr.instIdx = instance;

349  indexInstance = ocAllocateState(ocStateIsAllocated, ocStateInstanceMapping, &instancePtr, NUMBER_OF_OC_PAL_INSTANCES);
350  ocState = &g_ocPalStatePtr[indexInstance];
351  DEV_ASSERT(ocState->nNumChannels == 0x0U);
352 
353  /* Initialize the internal context to default value */
354  ocState->nNumChannels = configPtr->nNumChannels;
355  for (index = 0U; index < OC_PAL_CHANNEL_MAX_NUM; index++)
356  {
357  ocState->channelConfigArray[index] = 0U;
358  ocState->ocChannelMode[index] = OC_DISABLE_OUTPUT;
359  ocState->enableNotification[index] = false;

361  ocState->channelCallbacks[index] = NULL;

364  /* Get common parameters from user */
365  commonParam.allowDebugMode = false;
366  commonParam.lowPowerMode = false;
367  commonParam.clkDivVal = ((extension_emios_for_oc_t *)(configPtr->extension))->clkDivVal;
368  commonParam.enableGlobalPrescaler = ((extension_emios_for_oc_t *)(configPtr->extension))->enableGlobalPrescaler;
369  commonParam.enableGlobalTimeBase = ((extension_emios_for_oc_t *)(configPtr->extension))->enableGlobalTimeBase;
370 
371  /* Initialize the global for a eMIOS group */
372  EMIOS_DRV_InitGlobal((uint8_t)instance,
373  &commonParam);
374 
375  /* Get the information from user configuration */
376  mcParam.mode = EMIOS_MODE_MCB_UP_COUNTER_INT_CLK;
377  mcParam.filterInput = EMIOS_INPUT_FILTER_BYPASS;
378  mcParam.filterEn = false;
379  mcParam.triggerMode = EMIOS_TRIGGER_EDGE_ANY;
380 
381  for (index = 0U; index < configPtr->nNumChannels; index++)
382  {
383  mcParam.period = ((channel_extension_emios_for_oc_t *)(configPtr->outputChConfig[index].channelExtension))->period;
384  mcParam.internalPrescaler = ((channel_extension_emios_for_oc_t *)(configPtr->outputChConfig[index].channelExtension))->prescaler;
385  mcParam.internalPrescalerEn = true;
386 
387  switch (((channel_extension_emios_for_oc_t *)(configPtr->outputChConfig[index].channelExtension))->timebase)
388  {
389  case OC_BUS_SEL_A:
390  /* Set channel to use as a time base */
391  channel = 23U;
392  timeBaseSelection = EMIOS_BUS_SEL_A;
393  break;
394  case OC_BUS_SEL_B:
395  /* Set channel to use as a time base */
396  channel = 0U;
397  timeBaseSelection = EMIOS_BUS_SEL_BCDE;
398  break;
399  case OC_BUS_SEL_C:
400  /* Set channel to use as a time base */
401  channel = 8U;
402  timeBaseSelection = EMIOS_BUS_SEL_BCDE;
403  break;
404  case OC_BUS_SEL_D:
405  /* Set channel to use as a time base */
406  channel = 16U;
407  timeBaseSelection = EMIOS_BUS_SEL_BCDE;
408  break;
409  case OC_BUS_SEL_E:
410  /* Set channel to use as a time base */
411  channel = 24U;
412  timeBaseSelection = EMIOS_BUS_SEL_BCDE;
413  break;
414  case OC_BUS_SEL_F:
415  /* Set channel to use as a time base */
416  channel = 22U;
417  timeBaseSelection = EMIOS_BUS_SEL_F;
418  break;
419  case OC_BUS_SEL_INTERNAL:
420  /* Set channel to use as a time base */
421  channel = configPtr->outputChConfig[index].hwChannelId;
422  timeBaseSelection = EMIOS_BUS_SEL_INTERNAL;
423  break;
424  default:
425  /* Do nothing */
426  break;
427  }
428 
429  /* Initialize the counter mode */
430  status = EMIOS_DRV_MC_InitCounterMode((uint8_t)instance,
431  channel,
432  &mcParam);
433  if(STATUS_SUCCESS != status)
434  {
435  break;
436  }
437  channel = configPtr->outputChConfig[index].hwChannelId;
438  ocParam.mode = EMIOS_MODE_SAOC;
439  ocParam.timebase = timeBaseSelection;
440  ocParam.matchLeadingEdgeValue = (uint32_t)(configPtr->outputChConfig[index].comparedValue);
441  ocParam.matchTrailingEdgeValue = 0U;
442 
443  /* Check the operation mode of channel output */
444  if (OC_CLEAR_ON_MATCH == configPtr->outputChConfig[index].chMode)
445  {
446  ocParam.outputActiveMode = EMIOS_OUTPUT_ACTIVE_LOW;
447  }
448  else if (OC_SET_ON_MATCH == configPtr->outputChConfig[index].chMode)
449  {
450  ocParam.outputActiveMode = EMIOS_OUTPUT_ACTIVE_HIGH;
451  }
452  else
453  {
454  ocParam.outputActiveMode = EMIOS_OUTPUT_ACTIVE_TOGGLE;
455  }
456 
457  /* Initialize the output compare mode for each channel */
458  status = EMIOS_DRV_OC_InitOutputCompareMode((uint8_t)instance,
459  channel,
460  &ocParam);
461 
462  /* Enable EMIOS interrupt */
463  EMIOS_DRV_ChannelEnableInterruptRequest((uint8_t)instance, channel);
464  INT_SYS_EnableIRQ(emiosIrq[instance][(channel>>1U)]);
465 
466  /* Store some needed information into state structure */
467  ocState->channelConfigArray[index] = channel;
468  ocState->ocChannelMode[channel] = configPtr->outputChConfig[index].chMode;
469  ocState->enableNotification[channel] = true;
470  ocState->channelCallbackParams[channel] = configPtr->outputChConfig[index].channelCallbackParams;
471  ocState->channelCallbacks[channel] = configPtr->outputChConfig[index].channelCallbacks;
472  }
473 
474  /* Enable the global eMIOS to start counter */
475  EMIOS_DRV_EnableGlobalEmios((uint8_t)instance);
476 
477  return status;
478 }

536  ETIMER_DRV_Init(etimerInstance);

538  /* default configuration */
539  etimer_user_channel_config_t userChannelConfig = {

541  .coreSettings = ETIMER_COUNTER_NORMAL,
542  .inputFilter =

578  ETIMER_CPTMODE_DISABLED,

588  maskChannel |= (uint16_t)(1U << channel);
589

591  userChannelConfig.primaryInput.source = ((channel_extension_etimer_for_oc_t *)configPtr->outputChConfig[index].channelExtension)->primaryInput.source;
592  userChannelConfig.primaryInput.polarity = ((channel_extension_etimer_for_oc_t *)configPtr->outputChConfig[index].channelExtension)->primaryInput.polarity;
593

596  userChannelConfig.outputPin.enable = ((channel_extension_etimer_for_oc_t *)configPtr->outputChConfig[index].channelExtension)->outputPin.enable;
597

599  userChannelConfig.compareValues[0] = configPtr->outputChConfig[index].comparedValue;
600 
601  /* Check the operation mode of channel output */
602  if (OC_CLEAR_ON_MATCH == configPtr->outputChConfig[index].chMode)
603  {
604  userChannelConfig.compareOutputControl = ETIMER_OUTMODE_COMPARE_CLEAR;
605  }
606  else if (OC_SET_ON_MATCH == configPtr->outputChConfig[index].chMode)
607  {
608  userChannelConfig.compareOutputControl = ETIMER_OUTMODE_COMPARE_SET;
609  }
610  else if (OC_TOGGLE_ON_MATCH == configPtr->outputChConfig[index].chMode)
611  {
612  userChannelConfig.compareOutputControl = ETIMER_OUTMODE_COMPARE_TOGGLE;
613  }
614  else
615  {
616  userChannelConfig.compareOutputControl = ETIMER_OUTMODE_SOFTWARE;
617  maskChannel &= ~((uint16_t)(1U << channel));
618  /* disable the output pin, go to tri-state mode */
619  ETIMER_DRV_OutputPinDisable((uint16_t)etimerInstance, channel);
620  }

622  ETIMER_DRV_InitChannel((uint16_t)etimerInstance, channel, &userChannelConfig);

625 
626  /* Enable ETIMER channel interrupt and install an ISR */
627  IRQn_Type etimer_ch_irq_id = ETIMER_DRV_GetInterruptNumber((uint16_t)etimerInstance,(etimer_irq_vector_t)channel);
628  INT_SYS_EnableIRQ(etimer_ch_irq_id);
629

632  ocState->ocChannelMode[channel] = configPtr->outputChConfig[index].chMode;
633  ocState->enableNotification[channel] = true;
634  ocState->channelCallbackParams[channel] = configPtr->outputChConfig[index].channelCallbackParams;
635  ocState->channelCallbacks[channel] = configPtr->outputChConfig[index].channelCallbacks;
636  }
637 
638  /* Start the channel counting */
639  ETIMER_DRV_StartTimerChannels((uint16_t)etimerInstance, maskChannel);
640 
641  return STATUS_SUCCESS;
642 }

671  break;
672  }
673 
674  return STATUS_SUCCESS;
675 }

925 /*FUNCTION**********************************************************************

927  * Function Name : OC_FLEXPWM_SetOutputMode
928  * Description : This function will configure channel mode of output signal.

930  *END**************************************************************************/
931 static status_t OC_FLEXPWM_SetOutputMode(uint32_t instance,
932  uint8_t channel,
933  oc_option_mode_t channelMode)
934 {
935  DEV_ASSERT(instance < FlexPWM_INSTANCE_COUNT);
936  oc_pal_state_t * ocState;
937  oc_instance_t instancePtr;
938  uint8_t index = 0U;
939  uint16_t countValue1 = FLEXPWM_DRV_GetPeriod(instance, (flexpwm_module_t)channel);
940  uint16_t countValue2 = FLEXPWM_DRV_GetCmpRegValue(instance, (flexpwm_module_t)channel, FlexPwmVAL2);
941  uint16_t countValue3 = FLEXPWM_DRV_GetCmpRegValue(instance, (flexpwm_module_t)channel, FlexPwmVAL3);
942  uint16_t countValue4 = 0U;
943  uint16_t countValue5 = 0U;
944  uint16_t interruptMask = (uint16_t)FLEXPWM_CMP_ALL_VAL_INT_ENABLE;
945  instancePtr.instType = OC_INST_TYPE_FLEXPWM;
946  instancePtr.instIdx = instance;
947  index = FindOcState(&instancePtr);
948  ocState = &g_ocPalStatePtr[index];
949  oc_option_mode_t prvChMode = ocState->ocChannelMode[channel];
950 
951  /* Disable FLEXPWM channel interrupt and install an ISR */
952  FLEXPWM_DRV_DisableModuleInterrupts(instance,
953  (flexpwm_module_t)channel,
954  interruptMask);
955  switch (channelMode)
956  {
957  case OC_TOGGLE_ON_MATCH:
958  /* Set the pin to toggle output signal */
959  if(prvChMode == OC_CLEAR_ON_MATCH)
960  {
961  countValue2 = countValue3;
962  countValue3 += (countValue1 / 2U);
963  }
964  else
965  {
966  countValue3 = countValue2 + (countValue1 / 2U);
967  }
968  interruptMask = ((uint16_t)(((uint16_t)FLEXPWM_CMP_VAL2_INT_ENABLE) | ((uint16_t)FLEXPWM_CMP_VAL3_INT_ENABLE)) |
969  (uint16_t)(((uint16_t)FLEXPWM_CMP_VAL4_INT_ENABLE) | ((uint16_t)FLEXPWM_CMP_VAL5_INT_ENABLE)));
970  break;
971  case OC_CLEAR_ON_MATCH:
972  /* Set the output pin as high level */
973  if(prvChMode != OC_CLEAR_ON_MATCH)
974  {
975  countValue3 = countValue2;
976  }
977  countValue2 = countValue1 + 1U;
978  interruptMask = (((uint16_t)FLEXPWM_CMP_VAL2_INT_ENABLE) | ((uint16_t)FLEXPWM_CMP_VAL4_INT_ENABLE));
979  break;
980  case OC_SET_ON_MATCH:
981  /* Set the output pin as low level */
982  if(prvChMode == OC_CLEAR_ON_MATCH)
983  {
984  countValue2 = countValue3;
985  }
986  countValue3 = countValue1 + 1U;
987  interruptMask = (((uint16_t)FLEXPWM_CMP_VAL3_INT_ENABLE) | ((uint16_t)FLEXPWM_CMP_VAL5_INT_ENABLE));
988  break;
989  default:
990  /* Disable the output pin, go to tri-state mode */
991  countValue2 = 0U;
992  countValue3 = 0U;
993  break;
994  }
995  /* Update the count value */
996  countValue4 = countValue2;
997  countValue5 = countValue3;
998  OC_FLEXPWM_DRV_SetCountVal23(instance, channel, countValue2, countValue3);
999  OC_FLEXPWM_DRV_SetCountVal45(instance, channel, countValue4, countValue5);
1000 
1001  /* Reload register value */
1002  FLEXPWM_DRV_LoadCommands(instance, (uint32_t)(1UL << channel));
1003  /* Enable FLEXPWM channel interrupt and install an ISR */
1004  FLEXPWM_DRV_EnableModuleInterrupts(instance,
1005  (flexpwm_module_t)channel,
1006  interruptMask);
1007  return STATUS_SUCCESS;
1008 }
