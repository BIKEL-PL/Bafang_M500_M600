221  * Description : Calculate the smallest power of 2 greater than or equal to
222  * the given number
223  *
224  *END**************************************************************************/
225 static uint16_t SmallestPowerOfTwo(uint16_t number)
226 {
227  uint16_t helper;
228 
229  helper = number - 1u;
230  helper |= helper >> 1u;
231  helper |= helper >> 2u;
232  helper |= helper >> 4u;
233  helper |= helper >> 8u;
234  helper += 1u;
235  helper /= 8u;
236 
237  return helper;
238 }

369  * Description : Initializes a non-blocking master transfer
370  * Implements : SPI_MasterTransfer_Activity
371  *
372  *END**************************************************************************/
373 status_t SPI_MasterTransfer(const spi_instance_t * const instance, const void* txBuffer, void* rxBuffer, uint16_t numberOfFrames)
374 {
375  status_t status = STATUS_ERROR;
376 
377  /* Define SPI PAL over LPSPI */
378  #if defined(SPI_OVER_LPSPI)
379  if (instance->instType == SPI_INST_TYPE_LPSPI)
380  {
381  uint16_t bytesPerFrame;
382  uint8_t indexLpspi;
383 
384  indexLpspi = FindLpspiState(instance);
385 
386  if (indexLpspi != NO_OF_LPSPI_INSTS_FOR_SPI)
387  {
388  /* Calculate the smallest power of 2 greater than or equal to bitsPerFrame */
389  bytesPerFrame = SmallestPowerOfTwo(LpspiState[indexLpspi].bitsPerFrame);
390 
391  status = LPSPI_DRV_MasterTransfer(instance->instIdx, txBuffer, rxBuffer, numberOfFrames * bytesPerFrame);
392  }
393  }
394  else
395  #endif
396 
397  /* Define SPI PAL over FLEXIO */
398  #if defined(SPI_OVER_FLEXIO)
399  if (instance->instType == SPI_INST_TYPE_FLEXIO)
400  {
401  uint8_t indexFlexio;
402  indexFlexio = FindFlexioState(instance);
403 
404  if (indexFlexio != NO_OF_FLEXIO_INSTS_FOR_SPI)
405  {
406  uint16_t bytesPerFrame = FlexioSpiState[indexFlexio].transferSize;
407  status = FLEXIO_SPI_DRV_MasterTransfer(&(FlexioSpiState[indexFlexio]), txBuffer, rxBuffer, (uint32_t)((uint32_t)bytesPerFrame * (uint32_t)numberOfFrames));
408  }
409  }
410  else
411  #endif
412 
413  /* Define SPI PAL over DSPI */
414  #if (defined(SPI_OVER_DSPI))
415  if ((instance->instType == SPI_INST_TYPE_SPI) || (instance->instType == SPI_INST_TYPE_DSPI))
416  {
417  uint16_t inst = GetDspiIndex(instance);
418 
419  status = DSPI_MasterTransfer((dspi_instance_t)inst, txBuffer, rxBuffer, numberOfFrames);
420  }
421  else
422  #endif
423  {
424  DEV_ASSERT(0);
425  }
426  return status;
427 }
