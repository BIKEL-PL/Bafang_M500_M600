37 /*FUNCTION**********************************************************************
38  *
39  * Function Name : FTM_DRV_InitPwmIndependentChannel
40  * Description : Configures the PWM signal for the independent channel.
41  *
42  *END**************************************************************************/
43 static void FTM_DRV_InitPwmIndependentChannel(uint32_t instance,
44  const ftm_pwm_param_t * param)
45 {
46  FTM_Type * ftmBase = g_ftmBase[instance];
47  uint8_t index = 0U;
48  uint8_t channelId = 0U;
49  uint8_t chnlPairNum = 0U;
50 
51  /* Configure independent PWM channels */
52  for (index = 0U; index < param->nNumIndependentPwmChannels; index++)
53  {
54  channelId = param->pwmIndependentChannelConfig[index].hwChannelId;
55  chnlPairNum = (uint8_t)(channelId >> 1U);
56  /* Configure POL bits for fail safe state */
57  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId, (bool)param->pwmIndependentChannelConfig[index].safeState);
58 
59  /* Configure polarity of the PWM signal taking into consideration POL and ELSA/ELSB */
60  if ((uint32_t)(param->pwmIndependentChannelConfig[index].safeState) == (uint32_t)(param->pwmIndependentChannelConfig[index].polarity))
61  {
62  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)1U);
63  }
64  else
65  {
66  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)2U);
67  }
68 
69  if (param->pwmIndependentChannelConfig[index].enableSecondChannelOutput)
70  {
71  /* Configure dead time, and enable complementary channel. */
72  FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, param->pwmIndependentChannelConfig[index].deadTime);
73  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId + 1U, (uint8_t)2U);
74  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, true);
75  if (param->pwmIndependentChannelConfig[index].secondChannelPolarity == FTM_MAIN_INVERTED)
76  {
77  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId + 1U, (bool)param->pwmIndependentChannelConfig[index].safeState);
78  }
79  else
80  {
81  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId + 1U, !((bool)param->pwmIndependentChannelConfig[index].safeState));
82  }
83  }
84  else
85  {
86  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, false);
87  }
88 
89  /* Disable combined mode. */
90  FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);
91  FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);
92  FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);
93  /* Set MSB and MSA bits*/
94  FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);
95  /* Configure fault mode */
96  FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED) ? true : false);
97  /* Enable sync control for channels*/
98  FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, true);
99  /* Enable the generation a trigger on chip module */
100  FTM_DRV_SetChnTriggerCmd(ftmBase, channelId, param->pwmIndependentChannelConfig[index].enableExternalTrigger);
101 
102  /* Write FTMn_PWMLOAD register to enable synchronized loading points for the given channel */
103  FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);
104  if (param->pwmIndependentChannelConfig[index].enableSecondChannelOutput)
105  {
106  FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId + 1U);
107  }
108  }
109 }

117 static void FTM_DRV_InitPwmCombinedChannel(uint32_t instance,
118  const ftm_pwm_param_t * param)
119 {
120  FTM_Type * ftmBase = g_ftmBase[instance];
121  uint8_t index = 0U;
122  uint8_t channelId = 0U;
123  uint8_t chnlPairNum = 0U;
124 
125  /* Configure combined PWM channels */
126  for (index = 0U; index < param->nNumCombinedPwmChannels; index++)
127  {
128  channelId = param->pwmCombinedChannelConfig[index].hwChannelId;
129  chnlPairNum = (uint8_t)(channelId >> 1U);
130  /* Check if the channel id is even number */
131  DEV_ASSERT((channelId % 2U) == 0U);
132 
133  /* Configure POL bits for fail safe state */
134  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId, (bool)param->pwmCombinedChannelConfig[index].mainChannelSafeState);
135 
136  /* Configure polarity of the PWM signal taking into consideration POL and ELSA/ELSB */
137  if ((uint32_t)(param->pwmCombinedChannelConfig[index].mainChannelSafeState) == (uint32_t)(param->pwmCombinedChannelConfig[index].mainChannelPolarity))
138  {
139  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)1U);
140  }
141  else
142  {
143  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)2U);
144  }
145 
146  FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);
147  /* Set MSB and MSA bits */
148  FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);
149 
150  /* Enable channel (n) output */
151  FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);
152  /* Configure channel n+1 if it necessary. */
153  if (param->pwmCombinedChannelConfig[index].enableSecondChannelOutput)
154  {
155  channelId = channelId + 1U;
156  /* Configure POL bits for fail safe state */
157  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId, (bool)param->pwmCombinedChannelConfig[index].secondChannelSafeState);
158  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)2U);
159  /* Configure polarity of the second channel relative to main channel polarity. */
160  if (param->pwmCombinedChannelConfig[index].secondChannelSafeState == param->pwmCombinedChannelConfig[index].mainChannelSafeState)
161  {
162  if(param->pwmCombinedChannelConfig[index].secondChannelPolarity == FTM_MAIN_DUPLICATED)
163  {
164  /* If dead time is enabled and COMPx = 0 the channel n+1 is automatically disabled. */
165  DEV_ASSERT(!(param->pwmCombinedChannelConfig[index].deadTime));
166  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, false);
167  }
168  else
169  {
170  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, true);
171  }
172  }
173  else
174  {
175  if(param->pwmCombinedChannelConfig[index].secondChannelPolarity == FTM_MAIN_DUPLICATED)
176  {
177  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, true);
178  }
179  else
180  {
181  /* If dead time is enabled and COMPx = 0 the channel n+1 is automatically disabled. */
182  DEV_ASSERT(!(param->pwmCombinedChannelConfig[index].deadTime));
183  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, false);
184  }
185  }
186  /* Enable channel (n+1) output */
187  FTM_DRV_EnablePwmChannelOutputs(ftmBase, (uint8_t)(channelId));
188  }
189 
190  /* Set fault control for the channel */
191  FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED) ? true : false);
192  /* Enable sync control for channels */
193  FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, true);
194  /* Enable the combine mode */
195  FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, true);
196  /* Configure the modified combine mode */
197  FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, param->pwmCombinedChannelConfig[index].enableModifiedCombine);
198  /* Configure dead time */
199  FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, param->pwmCombinedChannelConfig[index].deadTime);
200  /* Enable the generation a trigger on the channel (n) */
201  channelId = (uint8_t)(chnlPairNum << 1U);
202  FTM_DRV_SetChnTriggerCmd(ftmBase, channelId, param->pwmCombinedChannelConfig[index].enableExternalTrigger);
203  /* Enable the generation a trigger on the channel (n+1) */
204  channelId = channelId + 1U;
205  FTM_DRV_SetChnTriggerCmd(ftmBase, channelId, param->pwmCombinedChannelConfig[index].enableExternalTriggerOnNextChn);
206  }
207 }

209 /*FUNCTION**********************************************************************
210  *
211  * Function Name : FTM_DRV_InitPwmDutyCycleChannel
212  * Description : This function will update the duty cycle for the PWM signal
213  * at the initialization.
214  *
215  *END**************************************************************************/
216 static status_t FTM_DRV_InitPwmDutyCycleChannel(uint32_t instance,
217  const ftm_pwm_param_t * param)
218 {
219  uint8_t index = 0U;
220  uint8_t hwChannel = 0U;
221  status_t retVal = STATUS_SUCCESS;
222 
223  for (index = 0U; index < param->nNumIndependentPwmChannels; index++)
224  {
225  hwChannel = param->pwmIndependentChannelConfig[index].hwChannelId;
226  /* Write CnV registers and setup duty cycle and phase values */
227  retVal = FTM_DRV_UpdatePwmChannel(instance,
228  hwChannel,
229  FTM_PWM_UPDATE_IN_DUTY_CYCLE,
230  param->pwmIndependentChannelConfig[index].uDutyCyclePercent,
231  0U,
232  false);
233  }
234 
235  for (index = 0U; index < param->nNumCombinedPwmChannels; index++)
236  {
237  hwChannel = param->pwmCombinedChannelConfig[index].hwChannelId;
238  /* Write CnV registers and setup duty cycle and phase values */
239  retVal = FTM_DRV_UpdatePwmChannel(instance,
240  hwChannel,
241  FTM_PWM_UPDATE_IN_DUTY_CYCLE,
242  param->pwmCombinedChannelConfig[index].firstEdge,
243  param->pwmCombinedChannelConfig[index].secondEdge,
244  false);
245  }
246 
247  return retVal;
248 }

250 /*FUNCTION**********************************************************************
251  *
252  * Function Name : FTM_DRV_InitPwm
253  * Description : Configures duty cycle and frequency and starts outputting
254  * PWM on specified channels.
255  *
256  * Implements : FTM_DRV_InitPwm_Activity
257  *END**************************************************************************/
258 status_t FTM_DRV_InitPwm(uint32_t instance,
259  const ftm_pwm_param_t * param)
260 {
261  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
262  DEV_ASSERT(param != NULL);
263  status_t retVal = STATUS_SUCCESS;
264  uint8_t fltChannel = 0U;
265  uint8_t faultChnNum = 0U;
266  uint32_t tempInst = instance;
267  ftm_state_t * state = ftmStatePtr[instance];
268  FTM_Type * ftmBase = g_ftmBase[instance];
269 
270  if ((NULL != state) && (FTM_MODE_NOT_INITIALIZED == state->ftmMode))
271  {
272  /* Disable counter clock */
273  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);
274  /* Clear the overflow flag */
275  FTM_DRV_ClearTimerOverflow(ftmBase);
276  /* Disable write protection */
277  FTM_DRV_SetWriteProtectionCmd(ftmBase, false);
278  /* Configure FTM mode */
279  state->ftmMode = param->mode;
280  /* Configure independent PWM channels */
281  FTM_DRV_InitPwmIndependentChannel(instance, param);
282  /* Configure combined PWM channels */
283  FTM_DRV_InitPwmCombinedChannel(instance, param);
284  /* Set enable outputs to be set to Initial/default value */
285  FTM_DRV_SetInitChnOutputCmd(ftmBase, true);
286  /* Enable faults (if faults were configured) */
287  if ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED)
288  {
289  /* Configure PWM Output behavior */
290  FTM_DRV_SetPwmFaultBehavior(ftmBase, ((param->faultConfig)->pwmOutputStateOnFault) ? true : false);
291  /* Configure fault filter value */
292  FTM_DRV_SetFaultInputFilterVal(ftmBase, ((param->faultConfig)->faultFilterValue));
293  /* Check the FTM instances */
294  if (tempInst <= 3U)
295  {
296  faultChnNum = (uint8_t)FTM_FEATURE_FAULT_CHANNELS;
297  }
298  else
299  {
300  faultChnNum = (uint8_t)(FTM_FEATURE_FAULT_CHANNELS >> 1U);
301  }
302 
303  for (fltChannel = 0U; fltChannel < faultChnNum; fltChannel++)
304  {
305  if (true == (param->faultConfig)->ftmFaultChannelParam[fltChannel].faultChannelEnabled)
306  {
307  /* Enable fault channel */
308  FTM_DRV_SetFaultInputCmd(ftmBase, fltChannel, true);
309  /* Configure fault filter */
310  FTM_DRV_SetFaultInputFilterCmd(ftmBase,
311  fltChannel,
312  ((param->faultConfig)->ftmFaultChannelParam[fltChannel].faultFilterEnabled) ? true : false);
313  /* Configure fault outputs */
314  FTM_DRV_SetChnFaultInputPolarityCmd(ftmBase,
315  fltChannel,
316  (((param->faultConfig)->ftmFaultChannelParam[fltChannel].ftmFaultPinPolarity == FTM_POLARITY_HIGH)? true : false));
317  }
318  }
319 
320  /* Set fault interrupt */
321  if (true == ((param->faultConfig)->pwmFaultInterrupt))
322  {
323  FTM_DRV_SetFaultInt(ftmBase, true);
324  }
325 
326  /* Enable fault control */
327  FTM_DRV_SetFaultControlMode(ftmBase, (uint32_t)(param->faultConfig)->faultMode);
328  }
329 
330  /* Configure PWM mode: edge or center aligned */
331  FTM_DRV_SetCpwms(ftmBase, (param->mode == FTM_MODE_CEN_ALIGNED_PWM) ? true : false);
332  /* Calculate frequency of the give FTM hardware module - all channels will run at the same frequency */
333  state->ftmPeriod = FTM_DRV_ConvertFreqToPeriodTicks(instance, param->uFrequencyHZ);
334  /* Based on reference manual, in PWM mode CNTIN is to be set 0*/
335  FTM_DRV_SetCounterInitVal(ftmBase, 0U);
336  /* Write MOD register with the value of the period */
337  /* For center aligned mode MOD register should be divided by 2 */
338  /* For edge aligned mode period is determined by: MOD-CNTIN+1 */
339  if (param->mode == FTM_MODE_CEN_ALIGNED_PWM)
340  {
341  FTM_DRV_SetMod(ftmBase, (uint16_t)(state->ftmPeriod >> 1U));
342  }
343  else
344  {
345  FTM_DRV_SetMod(ftmBase, (uint16_t)(state->ftmPeriod - 1U));
346  }
347 
348  /* Update the duty cycle */
349  retVal = FTM_DRV_InitPwmDutyCycleChannel(instance, param);
350 
351  if (STATUS_SUCCESS == retVal)
352  {
353  /* Configure dead time for combine mode */
354  FTM_DRV_SetDeadtimeCount(ftmBase, param->deadTimeValue);
355  FTM_DRV_SetDeadtimePrescale(ftmBase, param->deadTimePrescaler);
356  FTM_DRV_Enable(ftmBase, true);
357  FTM_DRV_SetPwmSyncMode(ftmBase, true);
358  /* Set clock source to start counter */
359  FTM_DRV_SetClockSource(ftmBase, state->ftmClockSource);
360  }
361  else
362  {
363  /* Restore FTM mode if initialization fails */
364  state->ftmMode = FTM_MODE_NOT_INITIALIZED;
365  }
366  }
367  else
368  {
369  retVal = STATUS_ERROR;
370  }
371 
372  return retVal;
373 }

375 /*FUNCTION**********************************************************************
376  *
377  * Function Name : FTM_DRV_DeinitPwm
378  * Description : Stops all PWM channels.
379  *
380  * Implements : FTM_DRV_DeinitPwm_Activity
381  *END**************************************************************************/
382 status_t FTM_DRV_DeinitPwm(uint32_t instance)
383 {
384  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
385  FTM_Type * ftmBase = g_ftmBase[instance];
386  uint8_t channel;
387  uint8_t chnlPairNum;
388  ftm_state_t * state = ftmStatePtr[instance];
389  DEV_ASSERT(state != NULL);
390 
391  /* Stop the FTM counter */
392  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);
393  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++)
394  {
395  chnlPairNum = (uint8_t)(channel >> 1U);
396  /* Disable PWM mode in hardware */
397  FTM_DRV_SetChnCountVal(ftmBase, channel, 0U);
398  FTM_DRV_SetChnEdgeLevel(ftmBase, channel, 0U);
399  FTM_DRV_SetChnMSnBAMode(ftmBase, channel, 0U);
400  FTM_DRV_SetCpwms(ftmBase, false);
401  /* Configure polarity bit */
402  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channel, false);
403  FTM_DRV_DisablePwmChannelOutputs(ftmBase, channel);
404  /* Clear the PWM synchronization */
405  FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, false);
406  /* Clear combination for each pair of channels */
407  FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);
408  FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);
409  FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, false);
410  FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, false);
411  }
412 
413  /* Clear the dead-time pre-scaler and value */
414  FTM_DRV_SetExtDeadtimeValue(ftmBase, 0U);
415  FTM_DRV_SetDeadtimePrescale(ftmBase, FTM_DEADTIME_DIVID_BY_1);
416  FTM_DRV_SetDeadtimeCount(ftmBase, 0U);
417  /* Clear fault control register */
418  FTM_DRV_ClearFaultControl(ftmBase);
419  /* Disable fault interrupt */
420  FTM_DRV_SetFaultInt(ftmBase, false);
421  /* Disable fault control */
422  FTM_DRV_SetFaultControlMode(ftmBase, (uint32_t)FTM_FAULT_CONTROL_DISABLED);
423  /* Clear the module value of the registers */
424  FTM_DRV_SetMod(ftmBase, 0U);
425  FTM_DRV_SetCounter(ftmBase, 0U);
426  state->ftmMode = FTM_MODE_NOT_INITIALIZED;
427 
428  return STATUS_SUCCESS;
429 }

431 /*FUNCTION**********************************************************************
432  *
433  * Function Name : FTM_DRV_UpdatePwmChannel
434  * Description : This function will update the duty cycle of PWM output.
435  * - If the type of update in the duty cycle, this function will convert the input parameters representing
436  * frequency in Hz to a period value in ticks needed by the hardware timer. Period is calculated depending
437  * on the operating mode of the FTM module and is stored in internal state structure.
438  * firstEdge and secondEdge can have value between 0 - FTM_MAX_DUTY_CYCLE(0 = 0% duty
439  * and FTM_MAX_DUTY_CYCLE = 100% duty). secondEdge value is used only whenFTM module is used in PWM combine mode.
440  * - If the type of update in ticks, this function will get value in ticks to the hardware timer.
441  * firstEdge and secondEdge variables can have value between 0 and ftmPeriod is stored in the state structure.
442  * - in the modified combine mode, the firstEdge parameter is fixed value and only can modify the secondEdge variables
443  * which is the initial value in the channel (n+1) match edge when the FTM counter has been ran.
444  *
445  * Implements : FTM_DRV_UpdatePwmChannel_Activity
446  *END**************************************************************************/
447 status_t FTM_DRV_UpdatePwmChannel(uint32_t instance,
448  uint8_t channel,
449  ftm_pwm_update_option_t typeOfUpdate,
450  uint16_t firstEdge,
451  uint16_t secondEdge,
452  bool softwareTrigger)
453 {
454  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
455  DEV_ASSERT(channel < FEATURE_FTM_CHANNEL_COUNT);
456  uint16_t hwFirstEdge = 0U;
457  uint16_t hwSecondEdge = 0U;
458  uint16_t ftmPeriod = 0U;
459  uint8_t chnlPairNum = (uint8_t)(channel >> 1U);
460  FTM_Type * ftmBase = g_ftmBase[instance];
461  ftm_state_t * state = ftmStatePtr[instance];
462  status_t retStatus = STATUS_SUCCESS;
463 
464  /* Get the newest period in the MOD register */
465  ftmPeriod = FTM_DRV_GetMod(ftmBase);
466  /* Check the mode operation in FTM module */
467  if (state->ftmMode == FTM_MODE_CEN_ALIGNED_PWM)
468  {
469  ftmPeriod = (uint16_t)(ftmPeriod << 1U);
470  }
471  else if ((state->ftmMode == FTM_MODE_EDGE_ALIGNED_PWM) || (state->ftmMode == FTM_MODE_EDGE_ALIGNED_PWM_AND_INPUT_CAPTURE))
472  {
473  ftmPeriod = (uint16_t)(ftmPeriod + 1U);
474  }
475  else
476  {
477  retStatus = STATUS_ERROR;
478  }
479 
480  /* Check the type of update for PWM */
481  if (FTM_PWM_UPDATE_IN_DUTY_CYCLE == typeOfUpdate)
482  {
483  if ((firstEdge <= FTM_MAX_DUTY_CYCLE) && (secondEdge <= FTM_MAX_DUTY_CYCLE))
484  {
485  /* Calculate DutyCycle based of the previously calculated frequency*/
486  /* For greater resolution the DutyCycle values are in the range [0. FTM_MAX_DUTY_CYCLE]
487  * where 0 = 0% or PWM always at Low and FTM_MAX_DUTY_CYCLE = 100% or PWM always HIGH;
488  * a value of 0x4000 is equivalent of 50% DutyCycle. */
489  hwFirstEdge = (uint16_t)((ftmPeriod * firstEdge) >> FTM_DUTY_TO_TICKS_SHIFT);
490  hwSecondEdge = (uint16_t)((ftmPeriod * secondEdge) >> FTM_DUTY_TO_TICKS_SHIFT);
491  /* adjust DutyCycle if 100% value is to be achieved. */
492  if (FTM_MAX_DUTY_CYCLE == firstEdge)
493  {
494  /* If expected duty is 100% then increase by 1 the value that is to be written
495  * to Hardware so it will exceed value of period */
496  hwFirstEdge = (uint16_t)(hwFirstEdge + 1U);
497  }
498  }
499  else
500  {
501  retStatus = STATUS_ERROR;
502  }
503  }
504  else
505  {
506  if ((firstEdge <= ftmPeriod) && (secondEdge <= ftmPeriod))
507  {
508  hwFirstEdge = firstEdge;
509  hwSecondEdge = secondEdge;
510  }
511  else
512  {
513  retStatus = STATUS_ERROR;
514  }
515  }
516 
517  if (STATUS_SUCCESS == retStatus)
518  {
519  if (true == FTM_DRV_GetDualChnCombineCmd(ftmBase, chnlPairNum))
520  {
521  if (true == FTM_DRV_GetDualChnMofCombineCmd(ftmBase, chnlPairNum))
522  {
523  /* Check the clock source for FTM counter is disabled or not */
524  if (FTM_DRV_GetClockSource(ftmBase) == 0U)
525  {
526  FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)(chnlPairNum * 2U), hwFirstEdge);
527  }
528  }
529  else
530  {
531  FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)(chnlPairNum * 2U), hwFirstEdge);
532  }
533 
534  /* Modify the initial value in the channel (n+1) match edge */
535  FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)((chnlPairNum * 2U) + 1U), hwSecondEdge);
536  }
537  else
538  {
539  /* Channel value is divided by 2 for up down counter mode to keep same duty */
540  if (true == FTM_DRV_GetCpwms(ftmBase))
541  {
542  FTM_DRV_SetChnCountVal(ftmBase, channel, (uint16_t)(hwFirstEdge >> 1U));
543  }
544  else
545  {

561 /*FUNCTION**********************************************************************
562  *
563  * Function Name : FTM_DRV_UpdatePwmPeriod
564  * Description : This function will update the new period in the frequency or
565  * in the counter value into mode register which modify the period of PWM signal
566  * on the channel output.
567  * - If the type of update in the duty cycle which is reused in FTM_DRV_UpdatePwmChannel
568  * function to convert the newValue parameters representing frequency in Hz to
569  * a period value to update the MOD register. The newValue parameter must be value
570  * between 1U and maximum is the frequency of the FTM counter.
571  * - If the type of update in ticks, this function will get value in counting to
572  * the MOD register. The newValue parameter must be value between 1U and 0xFFFFU
573  *
574  * Implements : FTM_DRV_UpdatePwmPeriod_Activity
575  *END**************************************************************************/
576 status_t FTM_DRV_UpdatePwmPeriod(uint32_t instance,
577  ftm_pwm_update_option_t typeOfUpdate,
578  uint32_t newValue,
579  bool softwareTrigger)
580 {
581  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
582  DEV_ASSERT(newValue != 0U);
583  uint32_t ftmPeriod = 0U;
584  FTM_Type * ftmBase = g_ftmBase[instance];
585  ftm_state_t * state = ftmStatePtr[instance];
586  status_t retStatus = STATUS_SUCCESS;
587 
588  /* Check the type of update for period in PWM mode */
589  if (FTM_PWM_UPDATE_IN_TICKS == typeOfUpdate)
590  {
591  ftmPeriod = newValue;
592  }
593  else
594  {
595  if (newValue <= state->ftmSourceClockFrequency)
596  {
597  ftmPeriod = (uint32_t)FTM_DRV_ConvertFreqToPeriodTicks(instance, newValue);
598  }
599  else
600  {
601  retStatus = STATUS_ERROR;
602  }
603  }
604 
605  if (STATUS_SUCCESS == retStatus)
606  {
607  /* Check the ftmPeriod is invalid */
608  DEV_ASSERT(ftmPeriod <= 0xFFFFU);
609  /* Check the signal operation in which PWM mode */
610  DEV_ASSERT((FTM_MODE_CEN_ALIGNED_PWM == state->ftmMode) || (FTM_MODE_EDGE_ALIGNED_PWM == state->ftmMode) || (FTM_MODE_EDGE_ALIGNED_PWM_AND_INPUT_CAPTURE == state->ftmMode));
611  if (FTM_MODE_CEN_ALIGNED_PWM == state->ftmMode)
612  {
613  ftmPeriod = (ftmPeriod >> 1U);
614  }
615  else
616  {
617  DEV_ASSERT(ftmPeriod > 0U);
618  ftmPeriod = (ftmPeriod - 1U);
619  }
620  /* Set the new modulo value into MOD register */
621  FTM_DRV_SetMod(ftmBase, (uint16_t)(ftmPeriod));
622  /* Software trigger is generated to change MOD registers */
623  FTM_DRV_SetSoftwareTriggerCmd(ftmBase, softwareTrigger);
624 
625  /* Store the PWM period in the state structure */
626  state->ftmPeriod = (uint16_t)ftmPeriod;
627  }
628 
629  return retStatus;
630 }

752 /*FUNCTION**********************************************************************
753  *
754  * Function Name : FTM_PWM_DRV_IrqHandler
755  * Description : This function is used to workaround an errata which the safe state
756  * is not removed from channel outputs after fault condition ends if SWOCTRL is being
757  * used to control the pin. The FTM_MODE[FAULTM] should be configured for manual fault
758  * clearing (0b10)
759  * This function must be used in the TOF interrupt handler when a fault is detected to
760  * ensure that the outputs return to the value configured by FTM_SWOCTR
761  *
762  * Implements : FTM_PWM_DRV_IrqHandler_Activity
763  *END**************************************************************************/
764 void FTM_PWM_DRV_IrqHandler(uint32_t instance,
765  uint32_t chnOutCtrlVal)
766 {
767  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
768  DEV_ASSERT(chnOutCtrlVal <= 0xFFu);
769  FTM_Type * ftmBase = g_ftmBase[instance];
770 
771  /* 1. Check the value of FTM_FMS[FAULTF] */
772  if (FTM_DRV_GetDetectedFaultInput(ftmBase) == true)
773  {
774  faultDetection = true;
775  /* 2. Write the FTM_OUTMASK register to set the bit */
776  FTM_DRV_SetOutmaskReg(ftmBase, chnOutCtrlVal);
777  /* 3. Clear fault conditions by reading the FTM_FMS register and then writing FTM_FMS with all zeroes */
778  FTM_DRV_ClearFaultsIsr(ftmBase);
779  /* 4. Clear the FTM_SC[TOF] bit by reading the FTM_SC register, then writing a 0 to FTM_SC[TOF] */
780  FTM_DRV_ClearTimerOverflow(ftmBase);
781  /* 5. Exit the interrupt handler to skip following steps */
782  }
783  else if (faultDetection == true)
784  {
785  /* 6. Clear the FTM_SWOCTRL by writing all zeroes to it */
786  FTM_DRV_SetAllChnSoftwareCtrlCmd(ftmBase, 0x00U);
787  FTM_DRV_SetAllChnSoftwareCtrlVal(ftmBase, 0x00U);
788  /* 7. Write FTM_SWOCTRL with the desired value again */
789  uint8_t u8chnOutCtrlVal = (uint8_t)(chnOutCtrlVal & 0xFFu);
790  FTM_DRV_SetAllChnSoftwareCtrlCmd(ftmBase, u8chnOutCtrlVal);
791  FTM_DRV_SetAllChnSoftwareCtrlVal(ftmBase, u8chnOutCtrlVal);
792  /* 8. Clear the FTM_OUTMASK bits that were set in step 2 */
793  FTM_DRV_SetOutmaskReg(ftmBase, 0x00U);
794  /* 9. Clear the fault variable that was set in step 1 when the fault condition was originally detected */
795  faultDetection = false;
796  /* 10. Clear the FTM_SC[TOF] bit by reading the FTM_SC register, then writing a 0 to FTM_SC[TOF] */
797  FTM_DRV_ClearTimerOverflow(ftmBase);
798  }
799  else
800  {
801  /* Nothing to do */
802  }
803 }
