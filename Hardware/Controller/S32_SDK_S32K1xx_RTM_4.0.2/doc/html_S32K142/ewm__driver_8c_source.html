52 /*FUNCTION**********************************************************************
53  *
54  * Function Name : EWM_DRV_Init
55  * Description : This function initializes the EWM instance to a specified
56  * state
57  *
58  * Implements : EWM_DRV_Init_Activity
59  *END**************************************************************************/
60 status_t EWM_DRV_Init(uint32_t instance, const ewm_init_config_t * config)
61 {
62  DEV_ASSERT(instance < EWM_INSTANCE_COUNT);
63  DEV_ASSERT(config != NULL);
64 
65  /* Return status variable */
66  status_t statusCode = STATUS_SUCCESS;
67  /* Flag to store if the module is enabled */
68  bool isModuleEnabled;
69  uint8_t tempValue = 0U;
70  /* Base pointer */
71  EWM_Type * base = s_ewmBase[instance];
72 
73  /* Get the enablement status of the module */
74  isModuleEnabled = EWM_IsEnabled(base);
75  /* Check if the EWM instance is already enabled or if the windows values are not correct */
76  if ((isModuleEnabled == true) || (config->compareHigh <= config->compareLow) ||
77  (config->compareHigh > FEATURE_EWM_CMPH_MAX_VALUE))
78  {
79  /* If conditions are met change the status code to error */
80  statusCode = STATUS_ERROR;
81  }
82  else
83  {
84  /* Set clock prescaler */
85  EWM_SetPrescaler(base, config->prescaler);
86  /* Set compare high and low values */
87  EWM_SetCompareHigh(base, config->compareHigh);
88  EWM_SetCompareLow(base, config->compareLow);
89 
90  /* Configure the Control register and enable the instance.
91  * Set the values that are not affected by the input pin
92  */
93  tempValue = (uint8_t)((uint8_t)EWM_CTRL_EWMEN(1U) | EWM_CTRL_INTEN(config->interruptEnable ? 1U : 0U));
94 
95  /* Depending how the input pin is configured set the values into the
96  * temporary variable
97  */
98  switch (config->assertLogic)
99  {
100  case EWM_IN_ASSERT_ON_LOGIC_ZERO:
101  tempValue |= EWM_CTRL_INEN_MASK; /* Input pin enabled, Input asserted on logic 0 */
102  break;
103  case EWM_IN_ASSERT_ON_LOGIC_ONE:
104  tempValue |= (uint8_t)(EWM_CTRL_INEN_MASK | EWM_CTRL_ASSIN_MASK); /* Input pin enabled, Input asserted on logic 1 */
105  break;
106  case EWM_IN_ASSERT_DISABLED:
107  default:
108  /* Input pin disabled */
109  break;
110  }
111 
112  /* Write the configuration into the Control register */
113  EWM_SetControl(base, tempValue);
114  }
115 
116  /* Return the status code */
117  return statusCode;
118 }

131 
132  /* Disable interrupts */

135  config->assertLogic = EWM_IN_ASSERT_ON_LOGIC_ZERO;

156 
157  /* Disable global IRQ's to avoid disturbing the refresh process */

172 ewm_in_assert_logic_t EWM_DRV_GetInputPinAssertLogic(uint32_t instance)
173 {
174  DEV_ASSERT(instance < EWM_INSTANCE_COUNT);
175 
176  /* Base pointer */
177  const EWM_Type * const base = s_ewmBase[instance];
178  /* Variable where to save the retrieved configuration */
179  ewm_in_assert_logic_t returnValue;
180 
181  /* Check if input pin is enabled */
182  if ((EWM_GetControl(base) & EWM_CTRL_INEN_MASK) != 0U)
183  {
184  /* If true get the assert logic into the returnValue variable */
185  returnValue = (((EWM_GetControl(base) & EWM_CTRL_ASSIN_MASK) != 0U)
186  ? EWM_IN_ASSERT_ON_LOGIC_ONE : EWM_IN_ASSERT_ON_LOGIC_ZERO);
187  }
188  else
189  {
190  /* Pin is disabled, return the corresponding value */
191  returnValue = EWM_IN_ASSERT_DISABLED;
192  }
193 
194  return returnValue;
195 }
