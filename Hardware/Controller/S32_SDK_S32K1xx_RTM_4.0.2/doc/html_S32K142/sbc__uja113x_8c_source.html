106 status_t UJA113X_SbcTransfer(uint32_t instance,
107  bool read,
108  uint8_t * txData,
109  uint8_t * rxData,
110  uint8_t size)
111 {
112  status_t retval = SBC_ERR_NA;
113 
114  DEV_ASSERT(instance < DRIVER_COUNT__);
115 
116  if (true == read)
117  {
118  /* Set LSB = read */
119  txData[0U] |= 0x01U;
120  }
121  else
122  {
123  /* Clear LSB = write */
124  txData[0U] &= 0xFEU;
125  }
126 
127  retval = LPSPI_DRV_MasterTransferBlocking(g_drivers[instance].lpspiInstance,
128  txData, rxData, size,
129  g_drivers[instance].timeout);
130  /*
131  * Check the communication status and first byte returned by SBC, the
132  * byte must match the register address
133  */
134  if (STATUS_SUCCESS == retval)
135  {
136  if (txData[0U] != rxData[0U])
137  {
138  retval = SBC_CMD_ERROR;
139  }
140  }
141 
142  return retval;
143 }
