112 static inline void LPUART_SetRxDataPolarity(LPUART_Type * base, bool polarity)
113 {
114  base->STAT = (base->STAT & (~FEATURE_LPUART_STAT_REG_FLAGS_MASK & ~LPUART_STAT_RXINV_MASK)) | \
115  ((polarity ? 1UL : 0UL) << LPUART_STAT_RXINV_SHIFT);
