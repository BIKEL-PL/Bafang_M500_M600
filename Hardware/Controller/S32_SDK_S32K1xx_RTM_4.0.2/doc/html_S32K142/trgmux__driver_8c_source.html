50 /*FUNCTION**********************************************************************
51  *
52  * Function Name : TRGMUX_DRV_Init
53  * Description : This function first resets the source triggers of all TRGMUX target modules
54  * to their default values, then configures the TRGMUX with all the user defined in-out mappings.
55  * If at least one of the target modules is locked, the function will not change any of the
56  * TRGMUX target modules and return an error code.
57  * This example shows how to set up the trgmux_user_config_t parameters and how to call the
58  * TRGMUX_DRV_Init() function with the required parameters:
59  * trgmux_user_config_t trgmuxConfig;
60  * trgmux_inout_mapping_config_t trgmuxInoutMappingConfig[] =
61  * {
62  * {TRGMUX_TRIG_SOURCE_TRGMUX_IN9, TRGMUX_TARGET_MODULE_DMA_CH0, false},
63  * {TRGMUX_TRIG_SOURCE_FTM1_EXT_TRIG, TRGMUX_TARGET_MODULE_TRGMUX_OUT4, true}
64  * };
65  * trgmuxConfig.numInOutMappingConfigs = 2;
66  * trgmuxConfig.inOutMappingConfig = trgmuxInoutMappingConfig;
67  * TRGMUX_DRV_Init(instance, &trgmuxConfig);
68  *
69  * Implements : TRGMUX_DRV_Init_Activity
70  *END**************************************************************************/
71 status_t TRGMUX_DRV_Init(const uint32_t instance,
72  const trgmux_user_config_t * const trgmuxUserConfig)
73 {
74  DEV_ASSERT(instance < TRGMUX_INSTANCE_COUNT);
75  DEV_ASSERT(trgmuxUserConfig != NULL);
76 
77  status_t status;
78  TRGMUX_Type * base = s_trgmuxBase[instance];
79  uint8_t count;
80 
81  /* Reset source triggers of all TRGMUX target modules to default. */
82  status = TRGMUX_Init(base);
83 
84  if (status == STATUS_SUCCESS)
85  {
86  /* Loop through all in-out mappings in the configuration and apply them in TRGMUX */
87  for (count = 0U; count < trgmuxUserConfig->numInOutMappingConfigs; count++)
88  {
89  TRGMUX_SetTrigSourceForTargetModule(base, trgmuxUserConfig->inOutMappingConfig[count].triggerSource,
90  trgmuxUserConfig->inOutMappingConfig[count].targetModule);
91  }
92 
93  /* Loop through all in-out mappings in the configuration and lock them if required */
94  for (count = 0U; count < trgmuxUserConfig->numInOutMappingConfigs; count++)
95  {
96  if (trgmuxUserConfig->inOutMappingConfig[count].lockTargetModuleReg)
97  {
98  TRGMUX_SetLockForTargetModule(base, trgmuxUserConfig->inOutMappingConfig[count].targetModule);
99  }
100  }
101  }
102 
103  return status;
104 }

127 /*FUNCTION**********************************************************************
128  *
129  * Function Name : TRGMUX_DRV_SetTrigSourceForTargetModule
130  * Description : This function configures a TRGMUX link between a source trigger and a target module,
131  * if the requested target module is not locked.
132  *
133  * Implements : TRGMUX_DRV_SetTrigSourceForTargetModule_Activity
134  *END**************************************************************************/
135 status_t TRGMUX_DRV_SetTrigSourceForTargetModule(const uint32_t instance,
136  const trgmux_trigger_source_t triggerSource,
137  const trgmux_target_module_t targetModule)
138 {
139  DEV_ASSERT(instance < TRGMUX_INSTANCE_COUNT);
140 
141  TRGMUX_Type * base = s_trgmuxBase[instance];
142  status_t status;
143  bool lock;
144 
145  lock = TRGMUX_GetLockForTargetModule(base, targetModule);
146 
147  if (lock == true)
148  {
149  status = STATUS_ERROR;
150  }
151  else
152  {
153  /* Configure link between trigger source and target module. */
154  TRGMUX_SetTrigSourceForTargetModule(base, triggerSource, targetModule);
155  status = STATUS_SUCCESS;
156  }
157 
158  return status;
159 }
