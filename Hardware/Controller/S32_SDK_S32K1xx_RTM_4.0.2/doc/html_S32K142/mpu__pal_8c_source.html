84 static void MPU_ConvertMpuRegionConfig(const mpu_region_config_t * regionConfig,
85  mpu_master_access_right_t * masterAccessRight,
86  mpu_user_config_t * configResult)
87 {
88  uint8_t masterIdx = 0U;
89  mpu_master_access_permission_t masterPrivilegeRight;
90 
91  /* Allocates region address */
92  configResult->startAddr = regionConfig->startAddr;
93  configResult->endAddr = regionConfig->endAddr;
94 
95  /* Allocates access rights for all masters */
96  for (masterIdx = 0U; masterIdx < FEATURE_MPU_MASTER_COUNT; masterIdx++)
97  {
98  masterPrivilegeRight = regionConfig->masterAccRight[masterIdx];
99 
100  /* Allocates master number */
101  masterAccessRight[masterIdx].masterNum = masterPrivilegeRight.masterNum;
102  /* Allocates access rights */
103  masterAccessRight[masterIdx].accessRight = (mpu_access_rights_t)masterPrivilegeRight.accessRight;
104  #if FEATURE_MPU_HAS_PROCESS_IDENTIFIER
105  /* Allocates process identifier */
106  masterAccessRight[masterIdx].processIdentifierEnable = ((regionConfig->processIdEnable
107  & (1UL << masterPrivilegeRight.masterNum))
108  != 0U);
109  #endif /* FEATURE_MPU_HAS_PROCESS_IDENTIFIER */
110  }
111 
112  /* Allocates master configuration in region */
113  configResult->masterAccRight = masterAccessRight;
114 
115 #if FEATURE_MPU_HAS_PROCESS_IDENTIFIER
116  /* Allocates process identifier */
117  configResult->processIdentifier = regionConfig->processIdentifier;
118  configResult->processIdMask = regionConfig->processIdMask;
119 #endif /* FEATURE_MPU_HAS_PROCESS_IDENTIFIER */
120 }

205 /*FUNCTION**********************************************************************
206  *
207  * Function Name : MPU_Init
208  * Description : Initializes memory protection unit by allocating regions
209  * and granting access rights for masters.
210  *
211  * Implements : MPU_Init_Activity
212  *END**************************************************************************/
213 status_t MPU_Init(const mpu_instance_t * const instance,
214  uint8_t regionCnt,
215  const mpu_region_config_t * configPtr)
216 {
217  DEV_ASSERT(instance != NULL);
218  DEV_ASSERT(configPtr != NULL);
219  DEV_ASSERT(regionCnt > 0U);
220 
221  status_t status = STATUS_UNSUPPORTED;
222  uint8_t regionIdx = 0U;
223 
224 #if defined(MPU_OVER_MPU)
225  if (instance->instType == MPU_INST_TYPE_MPU)
226  {
227  DEV_ASSERT(regionCnt <= MPU_RGD_COUNT);
228 
229  mpu_user_config_t mpuConfig[MPU_RGD_COUNT];
230  mpu_master_access_right_t mpuMasterConfig[MPU_RGD_COUNT * FEATURE_MPU_MASTER_COUNT];
231 
232  /* Allocates all region configuration */
233  for (regionIdx = 0U; regionIdx < regionCnt; regionIdx++)
234  {
235  MPU_ConvertMpuRegionConfig(&configPtr[regionIdx],
236  &mpuMasterConfig[regionIdx * FEATURE_MPU_MASTER_COUNT],
237  &mpuConfig[regionIdx]);
238  }
239  /* Initializes MPU module */
240  status = MPU_DRV_Init(instance->instIdx, regionCnt, mpuConfig);
241  }
242 #endif /* defined(MPU_OVER_MPU) */
243 
244 #if defined(MPU_OVER_SMPU)
245  if (instance->instType == MPU_INST_TYPE_SMPU)
246  {
247  DEV_ASSERT(regionCnt <= SMPU_RGD_COUNT);
248 
249  smpu_user_config_t smpuConfig[SMPU_RGD_COUNT];
250  smpu_master_access_right_t smpuMasterConfig[SMPU_RGD_COUNT * FEATURE_SMPU_MASTER_COUNT];
251 
252  /* Allocates all region configuration */
253  for (regionIdx = 0U; regionIdx < regionCnt; regionIdx++)
254  {
255  MPU_ConvertSmpuRegionConfig(&configPtr[regionIdx],
256  &smpuMasterConfig[regionIdx * FEATURE_SMPU_MASTER_COUNT],
257  &smpuConfig[regionIdx]);
258  }
259  /* Initializes SMPU module */
260  status = SMPU_DRV_Init(instance->instIdx, regionCnt, smpuConfig);
261  }
262 #endif /* defined(MPU_OVER_SMPU) */
263 
264  (void)regionIdx;
265  return status;
266 }

302 /*FUNCTION**********************************************************************
303  *
304  * Function Name : MPU_GetDefautRegionConfig
305  * Description : Gets default region configuration. Grants all access rights for masters;
306  * disable PID and cache inhibit; unlock region descriptor.
307  * Please note that the length of master array should be defined by number of masters supported by hardware.
308  *
309  * Implements : MPU_GetDefautRegionConfig_Activity
310  *END**************************************************************************/
311 status_t MPU_GetDefautRegionConfig(const mpu_instance_t * const instance,
312  mpu_master_access_permission_t * masterAccRight,
313  mpu_region_config_t * regionConfig)
314 {
315  DEV_ASSERT(masterAccRight != NULL);
316  DEV_ASSERT(regionConfig != NULL);
317 
318  status_t status = STATUS_UNSUPPORTED;
319  uint8_t masterIdx = 0U;
320 
321 #if defined(MPU_OVER_MPU)
322  if (instance->instType == MPU_INST_TYPE_MPU)
323  {
324  uint8_t mpuMasterNum[FEATURE_MPU_MASTER_COUNT] = FEATURE_MPU_MASTER;
325 
326  /* Gets default access rights for all masters */
327  for (masterIdx = 0U; masterIdx < FEATURE_MPU_MASTER_COUNT; masterIdx++)
328  {
329  masterAccRight[masterIdx].masterNum = mpuMasterNum[masterIdx];
330  if (mpuMasterNum[masterIdx] <= FEATURE_MPU_MAX_LOW_MASTER_NUMBER)
331  {
332  masterAccRight[masterIdx].accessRight = MPU_SUPERVISOR_RWX_USER_RWX;
333  }
334  else
335  {
336  masterAccRight[masterIdx].accessRight = MPU_RW;
337  }
338  }
339 
340  status = STATUS_SUCCESS;
341  }
342 #endif /* defined(MPU_OVER_MPU) */
343 
344 #if defined(MPU_OVER_SMPU)
345  if (instance->instType == MPU_INST_TYPE_SMPU)
346  {
347  uint8_t smpuMasterNum[FEATURE_SMPU_MASTER_COUNT] = FEATURE_SMPU_MASTER;
348 
349  /* Gets default access rights for all masters */
350  for (masterIdx = 0U; masterIdx < FEATURE_SMPU_MASTER_COUNT; masterIdx++)
351  {
352  masterAccRight[masterIdx].masterNum = smpuMasterNum[masterIdx];
353  masterAccRight[masterIdx].accessRight = MPU_RW_OR_SET_3;
354  }
355 
356  status = STATUS_SUCCESS;
357  }
358 #endif /* defined(MPU_OVER_SMPU) */
359 
360  /* Gets default region configuration */
361  regionConfig->startAddr = 0U;
362  regionConfig->endAddr = 0xFFFFFFFFU;
363  regionConfig->masterAccRight = masterAccRight;
364  regionConfig->processIdEnable = 0U;
365  regionConfig->processIdentifier = 0U;
366  regionConfig->processIdMask = 0U;
367  regionConfig->extension = NULL;
368 
369  (void)masterIdx;
370  return status;
371 }
