1758  FLEXCAN_CompleteTransfer(instance, mb_idx);

1772  }

2690

2692  numTq = clkFreq / ((tmpPresdiv + 1U) * bitrate);
2693  /* Compute the real bitrate resulted */
2694  tmpBitrate = clkFreq / ((tmpPresdiv + 1U) * numTq);
2695

2697  if ((numTq >= FLEXCAN_NUM_TQ_MIN) && (numTq < FLEXCAN_NUM_TQ_MAX))
2698  {
2699  /* Compute time segments based on the value of the sampling point */
2700  tSeg1 = (numTq * samplePoint / 100U) - 1U;
2701  tSeg2 = numTq - 1U - tSeg1;
2702 
2703  /* Adjust time segment 1 and time segment 2 */
2704  FLEXCAN_ProccessTSeg(&tSeg1, &tSeg2);
2705 
2706  tmpPseg2 = tSeg2 - 1U;
2707 
2708  /* Start from pseg1 = pseg2 and adjust until propseg is valid */
2709  tmpPseg1 = tmpPseg2;
2710  tmpPropseg = tSeg1 - tmpPseg1 - 2U;
2711 
2712  FLEXCAN_ProccessPSeg(&tmpPropseg, &tmpPseg1);
2713 
2714 
2715  if (((tSeg1 >= FLEXCAN_TSEG1_MAX) || (tSeg2 >= FLEXCAN_TSEG2_MAX) || (tSeg2 < FLEXCAN_TSEG2_MIN) || (tSeg1 < FLEXCAN_TSEG1_MIN)) ||
2716  ((tmpPropseg >= FLEXCAN_PROPSEG_MAX) || (tmpPseg1 >= FLEXCAN_PSEG1_MAX) || (tmpPseg2 < FLEXCAN_PSEG2_MIN) || (tmpPseg2 >= FLEXCAN_PSEG2_MAX)))
2717  {
2718  continue;
2719  }
2720 
2721  tmpSample = ((tSeg1 + 1U) * 100U) / numTq;
2722  dSample = FLEXCAN_CheckDsample(tmpSample , samplePoint);
2723  dBitrate = FLEXCAN_CheckdBitrate(tmpBitrate , bitrate);
2724 
2725  if ((dBitrate < dBitrateMin) ||
2726  ((dBitrate == dBitrateMin) && (dSample < dSampleMin)))
2727  {
2728  dSampleMin = dSample;
2729  dBitrateMin = dBitrate;
2730  pseg1 = tmpPseg1;
2731  pseg2 = tmpPseg2;
2732  presdiv = tmpPresdiv;
2733  propseg = tmpPropseg;
2734 
2735  if ((dBitrate == 0U) && (dSample <= 1U))
2736  {
2737  break;
2738  }
2739  }
2740  }
2741  }
2742

2744  timeSeg->phaseSeg2 = pseg2;
2745  timeSeg->preDivider = presdiv;
2746  timeSeg->propSeg = propseg;
2747  timeSeg->rJumpwidth = FLEXCAN_CheckJumpwidth(pseg1);
