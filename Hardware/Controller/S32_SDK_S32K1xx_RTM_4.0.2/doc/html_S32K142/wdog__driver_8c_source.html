62 /* Gets the frequency of the specified WDOG clock source. Only used at development
63  * time, when WDOG_DRV_Init checks if the needed clock sources are enabled. */
64 static uint32_t WDOG_DRV_GetClockSourceFreq(wdog_clk_source_t wdogClk)
65 {
66  uint32_t freq = 0;
67 
68  switch (wdogClk)
69  {
70  case WDOG_BUS_CLOCK:
71  (void)CLOCK_SYS_GetFreq(BUS_CLK, &freq);
72  break;
73  case WDOG_SIRC_CLOCK:
74  (void)CLOCK_SYS_GetFreq(SIRC_CLK, &freq);
75  break;
76  case WDOG_SOSC_CLOCK:
77  (void)CLOCK_SYS_GetFreq(SOSC_CLK, &freq);
78  break;
79  case WDOG_LPO_CLOCK:
80  (void)CLOCK_SYS_GetFreq(SIM_LPO_CLK, &freq);
81  break;
82  default:
83  /* Should not get here */
84  break;
85  }

95 /*FUNCTION**********************************************************************
96  *
97  * Function Name : WDOG_DRV_Init
98  * Description : initialize the WDOG driver
99  *
100  * Implements : WDOG_DRV_Init_Activity
101  *END**************************************************************************/
102 status_t WDOG_DRV_Init(uint32_t instance,
103  const wdog_user_config_t * userConfigPtr)
104 {
105  DEV_ASSERT(instance < WDOG_INSTANCE_COUNT);
106  DEV_ASSERT(userConfigPtr != NULL);
107  WDOG_Type * base = s_wdogBase[instance];
108  status_t status = STATUS_SUCCESS;
109  status_t statusClockSource = STATUS_SUCCESS;
110 
111 #ifdef DEV_ERROR_DETECT
112  uint32_t prevClockHz, crtClockHz;
113 
114  /* Check if the previous clock source and the configuration clock source
115  * are enabled(if not, the counter will not be incremented) */
116  prevClockHz = WDOG_DRV_GetClockSourceFreq(WDOG_GetClockSource(s_wdogBase[instance]));
117  crtClockHz = WDOG_DRV_GetClockSourceFreq(userConfigPtr->clkSource);
118 
119  if ((prevClockHz == 0U) || (crtClockHz == 0U))
120  {
121  statusClockSource = STATUS_ERROR;
122  }
123 #endif
124  /* If clock source disabled */
125  if (statusClockSource == STATUS_SUCCESS)
126  {
127  /* If window mode enabled and window value greater than or equal to the timeout value. Or timeout value is 0 */
128  if (((userConfigPtr->winEnable) && (userConfigPtr->windowValue >= userConfigPtr->timeoutValue)) \
129  || (userConfigPtr->timeoutValue <= FEATURE_WDOG_MINIMUM_TIMEOUT_VALUE))
130  {
131  status = STATUS_ERROR;
132  }
133  else
134  {
135  /* Configure the WDOG module */
136  status = WDOG_Config(base, userConfigPtr);
137  }
138 
139  if (status == STATUS_SUCCESS)
140  {
141  /* enable WDOG timeout interrupt */
142  INT_SYS_EnableIRQ(s_wdogIrqId[instance]);
143  }
144  }
145  else
146  {
147  status = STATUS_ERROR;
148  }
149 
150  return status;
151 }

153 /*FUNCTION**********************************************************************
154  *
155  * Function Name : WDOG_DRV_Deinit
156  * Description : De-initialize the WDOG driver
157  *
158  * Implements : WDOG_DRV_Deinit_Activity
159  *END**************************************************************************/
160 status_t WDOG_DRV_Deinit(uint32_t instance)
161 {
162  DEV_ASSERT(instance < WDOG_INSTANCE_COUNT);
163  WDOG_Type * base = s_wdogBase[instance];
164  status_t status = STATUS_SUCCESS;
165 
166  INT_SYS_DisableIRQGlobal();
167 
168  /* If allowed reconfigures WDOG */
169  if (WDOG_IsUpdateEnabled(base))
170  {
171  /* Disable WDOG timeout interrupt */
172  INT_SYS_DisableIRQ(s_wdogIrqId[instance]);
173 
174  /* Disable WDOG */
175  WDOG_Deinit(base);
176  }
177  else
178  {
179  status = STATUS_ERROR;
180  }
181 
182  /* Enable global interrupt */
183  INT_SYS_EnableIRQGlobal();
184 
185  return status;
186 }
