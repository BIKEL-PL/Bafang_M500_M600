89 static uint8_t PwmAllocateState(bool* isAllocated, uint32_t* instanceMapping, const pwm_instance_t * const instance, uint8_t numberOfinstances)
90 {
91  uint8_t i;
92  /* Allocate one of the FTM state structure for this instance */
93  for (i = 0;i < numberOfinstances;i++)
94  {
95  if (isAllocated[i] == false)
96  {
97  instanceMapping[i] = instance->instIdx;
98  isAllocated[i] = true;
99  break;
100  }
101  }
102  return i;
103 }

577  status_t status = STATUS_ERROR;
578 
579  #if (defined(PWM_OVER_FTM))

584  {

587  }

591  }
592  }

594  #endif
595 
596  #if (defined(PWM_OVER_EMIOS))

598  {
599  if ((eMIOS[instance->instIdx]->UC[channel].C & eMIOS_C_MODE_MASK) == (uint32_t)EMIOS_MODE_OPWMB_FLAGX1)
600  {
601  eMIOS[instance->instIdx]->UC[channel].B = duty;
602  }
603  else
604  {
605  eMIOS[instance->instIdx]->UC[channel].A = duty;
606  }
607  status = STATUS_SUCCESS;
608  }
609  else
610  #endif
611 
612  #if (defined (PWM_OVER_ETIMER))
613  if (instance->instType == PWM_INST_TYPE_ETIMER)
614  {
615  /* read current compare values */
616  uint16_t comp1;
617  uint16_t comp2;
618  ETIMER_DRV_GetCompareThresholdBuffered(instance->instIdx, channel, &comp1, &comp2);
619 
620  /* the duty cycle must be lower or equal the period */
621  uint16_t ton = comp1 + 1u;
622  uint16_t toff = comp2 + 1u;
623  uint16_t period = ton + toff;
624  if(duty <= period)
625  {
626  if (duty == 0u)
627  {
628  ton = 1u;
629  }
630  else if (period == duty)
631  {
632  ton = period - 1u;
633  }
634  else
635  {
636  ton = duty;
637  }
638  toff = period - ton;
639  comp1 = ton - 1u;
640  comp2 = toff - 1u;
641  /* set it */
642  ETIMER_DRV_SetCompareThresholdBuffered(instance->instIdx, channel, comp1, comp2);
643  /* all ok */
644  status = STATUS_SUCCESS;
645  }
646  else
647  {
648  /* duty is higher than period */
649  DEV_ASSERT(false);
650  }
651  }
652  else
653  #endif
654  {
655  DEV_ASSERT(false);
656  }
657 
658  return status;
659 }

669 status_t PWM_UpdatePeriod(const pwm_instance_t * const instance, uint8_t channel, uint32_t period)
670 {
671 
672  DEV_ASSERT(instance != NULL);
673 
674  status_t status = STATUS_ERROR;
675

680  (void)channel;
681  status = STATUS_SUCCESS;
682  }
683  else
684  #endif
685 
686  #if (defined(PWM_OVER_EMIOS))
687  if (instance->instType == PWM_INST_TYPE_EMIOS)
688  {
689  switch (((eMIOS[instance->instIdx]->UC[channel].C) & eMIOS_C_BSL_MASK) >> eMIOS_C_BSL_SHIFT)

691  case 0U:
692  eMIOS[instance->instIdx]->UC[BUS_A].A = period;
693  break;
694  case 1U:

697  eMIOS[instance->instIdx]->UC[BUS_B].A = period;
698  }
699  else if (channel <= 15U)
700  {
701  eMIOS[instance->instIdx]->UC[BUS_C].A = period;
702  }
703  else if (channel <= 23U)
704  {
705  eMIOS[instance->instIdx]->UC[BUS_D].A = period;
706  }
707  else if (channel <= 31U)
708  {
709  eMIOS[instance->instIdx]->UC[BUS_E].A = period;
710  }
711  else
712  {
713  /* Noting to do */
714  }
715  break;
716  case 2U:
717  eMIOS[instance->instIdx]->UC[BUS_F].A = period;
718  break;
719  default: /* Noting to do */
720  break;
721  }
722  status = STATUS_SUCCESS;
723  }
724  else
725  #endif
726 
727  #if (defined (PWM_OVER_ETIMER))
728  if (instance->instType == PWM_INST_TYPE_ETIMER)
729  {
730  /* read current compare values */
731  uint16_t comp1;
732  ETIMER_DRV_GetCompareThresholdBuffered(instance->instIdx, channel, &comp1, NULL);
733  uint16_t ton = comp1 + 1u;
734  /* the period must be higher than the duty (Ton)
735  * because the eTimer cannot generate a 100% duty signal
736  */
737  if(period > ton)
738  {
739  /* set it */
740  DEV_ASSERT((period - ton) <= 0xFFFFu); /* check if the Toff length fits in 16 bits */
741  uint16_t toff = period - ton;
742  uint16_t comp2 = toff - 1u;
743  /* set it */
744  ETIMER_DRV_SetCompareThresholdBuffered(instance->instIdx, channel, comp1, comp2);
745  /* all ok */
746  status = STATUS_SUCCESS;
747  }
748  else
749  {
750  /* duty is higher than period */
751  DEV_ASSERT(false);
752  }
753  }
754  else
755  #endif
756  {
757  DEV_ASSERT(false);
758  }
759  return status;
760 }
