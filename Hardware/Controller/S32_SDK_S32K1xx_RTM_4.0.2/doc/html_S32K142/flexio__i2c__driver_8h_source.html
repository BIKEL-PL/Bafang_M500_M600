62 /* Length of the DMA scatter-gather chain for Tx. Two blocks are needed, but as the fist one is
63  loaded directly in the TCD, only one block needs to be in RAM */
64 #define FLEXIO_I2C_DMA_TX_CHAIN_LENGTH (1U)
65 /* Length of the DMA scatter-gather chain for Rx. Six blocks are needed, but as the fist one is
66  loaded directly in the TCD, only five blocks needs to be in RAM */
67 #define FLEXIO_I2C_DMA_RX_CHAIN_LENGTH (5U)
68 /* Total size of the DMA scatter-gather buffer. Add 1 because STCD_SIZE already compensates for the
69  first stcd not being loaded in RAM */
70 #define FLEXIO_I2C_DMA_BUF_SIZE (STCD_SIZE(FLEXIO_I2C_DMA_TX_CHAIN_LENGTH + FLEXIO_I2C_DMA_RX_CHAIN_LENGTH + 1U))
