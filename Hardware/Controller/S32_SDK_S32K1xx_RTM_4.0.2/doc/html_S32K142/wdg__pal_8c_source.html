75  * Function Name : WDG_Init
76  * Description : Initialize the WDG PAL
77  *
78  * Implements : WDG_Init_Activity
79  *END**************************************************************************/
80 status_t WDG_Init(const wdg_instance_t * const instance,
81  const wdg_config_t * configPtr)
82 {
83  DEV_ASSERT(instance != NULL);
84  DEV_ASSERT(configPtr != NULL);
85 
86  uint32_t windowValue = 0U;
87  status_t status = STATUS_UNSUPPORTED;
88 
89  /* Calculate window value */
90  if (true == configPtr->winEnable)
91  {
92  DEV_ASSERT(configPtr->percentWindow <= 100U);
93 
94  windowValue = (uint32_t)(((uint64_t)configPtr->percentWindow * configPtr->timeoutValue) / 100U);
95  }
96 
97  /* Define WDG PAL over WDOG */
98 #if defined (WDG_OVER_WDOG)
99  if (WDG_INST_TYPE_WDOG == instance->instType)
100  {
101  /* Define structures for WDG PAL over WDOG */
102  wdog_user_config_t wdogConfig =
103  {
104  .clkSource = (wdog_clk_source_t)configPtr->clkSource,
105  .opMode =
106  {
107  .wait = configPtr->opMode.wait,
108  .stop = configPtr->opMode.stop,
109  .debug = configPtr->opMode.debug
110  },
111  .updateEnable = true,
112  .intEnable = configPtr->intEnable,
113  .winEnable = configPtr->winEnable,
114  .windowValue = (uint16_t)((configPtr->timeoutValue - windowValue) & 0xFFFFU),
115  .timeoutValue = (uint16_t)(configPtr->timeoutValue & 0xFFFFU),
116  .prescalerEnable = configPtr->prescalerEnable
117  };
118 
119  /* Initialize WDOG instance */
120  status = WDOG_DRV_Init(instance->instIdx, &wdogConfig);
121  }
122 #endif /* defined (WDG_OVER_WDOG) */
123 
124  /* Define WDG PAL over EWM */
125 #if defined (WDG_OVER_EWM)
126  if (WDG_INST_TYPE_EWM == instance->instType)
127  {
128  /* Define structures for WDG PAL over EWM */
129  ewm_init_config_t ewmConfig;
130 
131  /* Set value for EWM structure */
132  ewmConfig.interruptEnable = configPtr->intEnable;
133 
134  /* Set value for input pin asserts and clock prescaler */
135  if (NULL == configPtr->extension)
136  {
137  ewmConfig.assertLogic = EWM_IN_ASSERT_ON_LOGIC_ZERO;
138  ewmConfig.prescaler = 0xFFU;
139  }
140  else
141  {
142  ewmConfig.assertLogic = (ewm_in_assert_logic_t)((extension_ewm_for_wdg_t*)(configPtr->extension))->assertLogic;
143  ewmConfig.prescaler = ((extension_ewm_for_wdg_t*)(configPtr->extension))->prescalerValue;
144  }
145 
146  /* Set value for compare low register */
147  if (true == configPtr->winEnable)
148  {
149  ewmConfig.compareLow = (uint8_t)((configPtr->timeoutValue - windowValue) & 0xFFU);
150  }
151  else
152  {
153  ewmConfig.compareLow = FEATURE_EWM_CMPL_MIN_VALUE;
154  }
155 
156  /* Set value for compare high register */
157  if (configPtr->timeoutValue > FEATURE_EWM_CMPH_MAX_VALUE)
158  {
159  ewmConfig.compareHigh = FEATURE_EWM_CMPH_MAX_VALUE;
160  }
161  else
162  {
163  ewmConfig.compareHigh = (uint8_t)(configPtr->timeoutValue & 0xFFU);
164  }
165 
166  /* Initialize EWM instance */
167  status = EWM_DRV_Init(instance->instIdx, &ewmConfig);
168  }
169 #endif /* defined (WDG_OVER_EWM) */
170 
171  /* Define WDG PAL over SWT */
172 #if defined (WDG_OVER_SWT)
173  if (WDG_INST_TYPE_SWT == instance->instType)
174  {
175  /* Define structures for WDG PAL over SWT */
176  swt_user_config_t swtConfig =
177  {
178  .mapConfig = 0xFFU,
179  .invalidReset = false,
180  .winEnable = configPtr->winEnable,
181  .intEnable = configPtr->intEnable,
182  .stop = configPtr->opMode.stop,
183  .debug = configPtr->opMode.debug,
184  .serviceMode = SWT_FS_SEQ_MODE,
185  .lockConfig = SWT_UNLOCK,
186  .timeoutValue = configPtr->timeoutValue,
187  .windowValue = windowValue,
188  .initKey = 0U
189  };
190 
191  /* Initialize SWT instance */
192  status = SWT_DRV_Init(instance->instIdx, &swtConfig);
193  }
194 #endif /* defined (WDG_OVER_SWT) */
195 
196  return status;
197 }

431 /*FUNCTION**********************************************************************
432  *
433  * Function Name : WDG_GetCounter
434  * Description : Gets the value of the WDG PAL counter.
435  * Note that: Counter will be reset to timeout value if WDG PAL uses SWT.
436  * The counter will continue to run if WDG PAL uses WDOG.
437  *
438  * Implements : WDG_GetCounter_Activity
439  *END**************************************************************************/
440 status_t WDG_GetCounter(const wdg_instance_t * const instance,
441  uint32_t * value)
442 {
443  DEV_ASSERT(instance != NULL);
444 
445  status_t status = STATUS_UNSUPPORTED;
446 
447  /* Define WDG PAL over WDOG */
448 #if defined(WDG_OVER_WDOG)
449  if (WDG_INST_TYPE_WDOG == instance->instType)
450  {
451  /* Get counter value over WDOG module */
452  *value = WDOG_DRV_GetCounter(instance->instIdx);
453  status = STATUS_SUCCESS;
454  }
455 #endif /* defined (WDG_OVER_WDOG) */
456 
457  /* Define WDG PAL over SWT */
458 #if defined(WDG_OVER_SWT)
459  if (WDG_INST_TYPE_SWT == instance->instType)
460  {
461  /* Disable SWT */
462  status = SWT_DRV_StopTimer(instance->instIdx);
463  /* Get counter value of SWT */
464  if (STATUS_SUCCESS == status)
465  {
466  /* Get counter value over SWT module */
467  (void)SWT_DRV_GetCounterValue(instance->instIdx, value);
468  /* Enable SWT */
469  (void)SWT_DRV_StartTimer(instance->instIdx);
470  }
471  }
472 #endif /* defined (WDG_OVER_SWT) */
473 
474  /* Just eliminate warning if input parameters are not used*/
475  (void)instance;
476  (void)value;
477 
478  return status;
479 }
