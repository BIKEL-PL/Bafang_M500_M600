181  const ic_config_t * configPtr)
182 {
183  DEV_ASSERT(instance < IC_PAL_INSTANCE_MAX);
184  ic_pal_state_t * icState;
185  ic_instance_t instancePtr;
186  status_t status = STATUS_SUCCESS;
187  uint8_t index = 0U;
188  /* Table of base addresses for FTM instances. */
189  static FTM_Type * const s_ftmBase[FTM_INSTANCE_COUNT] = FTM_BASE_PTRS;
190  FTM_Type * base;
191  base = s_ftmBase[instance];
192 
193  /* Allocate one of the IC state structure for this instance */
194  instancePtr.instIdx = instance;
195  instancePtr.instType = IC_INST_TYPE_FTM;
196  index = FindIcState(&instancePtr);
197  icState = &g_icPalStatePtr[index];
198 
199  uint8_t channel = 0U;
200  ftm_pwm_sync_t sync;
201  ftm_user_config_t info;
202  ftm_state_t * ftmState = &g_ftmState[index];
203  ftm_input_param_t param;
204  ftm_input_ch_param_t channelConfig[FEATURE_FTM_CHANNEL_COUNT];
205 
206  /* Configure the synchronous to default */
207  sync.softwareSync = true;
208  sync.hardwareSync0 = false;
209  sync.hardwareSync1 = false;
210  sync.hardwareSync2 = false;
211  sync.maxLoadingPoint = true;
212  sync.minLoadingPoint = false;
213  sync.inverterSync = FTM_SYSTEM_CLOCK;
214  sync.outRegSync = FTM_SYSTEM_CLOCK;
215  sync.maskRegSync = FTM_SYSTEM_CLOCK;
216  sync.initCounterSync = FTM_SYSTEM_CLOCK;
217  sync.autoClearTrigger = false;
218  sync.syncPoint = FTM_UPDATE_NOW;
219 
220  /* Get some information from user configuration */
221  info.syncMethod = sync;
222  info.ftmMode = FTM_MODE_INPUT_CAPTURE;
223  info.ftmPrescaler = ((extension_ftm_for_ic_t *)(configPtr->extension))->ftmPrescaler;
224  info.ftmClockSource = ((extension_ftm_for_ic_t *)(configPtr->extension))->ftmClockSource;
225  info.BDMMode = FTM_BDM_MODE_11;
226  info.isTofIsrEnabled = false;
227  info.enableInitializationTrigger = false;
228 
229  /* Initializes the FTM driver */
230  status = FTM_DRV_Init(instance,
231  &info,
232  ftmState);
233  DEV_ASSERT(STATUS_SUCCESS == status);
234 
235  param.nNumChannels = configPtr->nNumChannels;
236  param.nMaxCountValue = MAX_COUNTER_VALUE;
237 
238  /* Configure a list of channels which are used */
239  for (index = 0U; index < configPtr->nNumChannels; index++)
240  {
241  /* Get the hardware channel ID */
242  channelConfig[index].hwChannelId = configPtr->inputChConfig[index].hwChannelId;
243  channel = channelConfig[index].hwChannelId;
244 
245  /* Check the type of measurement */
246  switch (configPtr->inputChConfig[index].inputCaptureMode)
247  {
248  case IC_TIMESTAMP_RISING_EDGE:
249  channelConfig[index].edgeAlignement = FTM_RISING_EDGE;
250  break;
251  case IC_TIMESTAMP_FALLING_EDGE:
252  channelConfig[index].edgeAlignement = FTM_FALLING_EDGE;
253  break;
254  case IC_TIMESTAMP_BOTH_EDGES:
255  channelConfig[index].edgeAlignement = FTM_BOTH_EDGES;
256  break;
257  case IC_MEASURE_RISING_EDGE_PERIOD:
258  channelConfig[index].measurementType = FTM_RISING_EDGE_PERIOD_MEASUREMENT;
259  break;
260  case IC_MEASURE_FALLING_EDGE_PERIOD:
261  channelConfig[index].measurementType = FTM_FALLING_EDGE_PERIOD_MEASUREMENT;
262  break;
263  case IC_MEASURE_PULSE_HIGH:
264  channelConfig[index].measurementType = FTM_PERIOD_ON_MEASUREMENT;
265  break;
266  case IC_MEASURE_PULSE_LOW:
267  channelConfig[index].measurementType = FTM_PERIOD_OFF_MEASUREMENT;
268  break;
269  default:
270  /* Do nothing */
271  break;
272  }
273 
274  /* Check the input capture operation mode */
275  if ((configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_RISING_EDGE) || \
276  (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_FALLING_EDGE) || \
277  (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_BOTH_EDGES))
278  {
279  channelConfig[index].inputMode = FTM_EDGE_DETECT;
280  channelConfig[index].measurementType = FTM_NO_MEASUREMENT;
281  }
282  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_DISABLE_OPERATION)
283  {
284  channelConfig[index].inputMode = FTM_NO_OPERATION;
285  channelConfig[index].edgeAlignement = FTM_NO_PIN_CONTROL;
286  channelConfig[index].measurementType = FTM_NO_MEASUREMENT;
287  }
288  else
289  {
290  /* Check the channel ID need to even number in the measurement mode */
291  DEV_ASSERT((channel % 2U) == 0U);
292  channelConfig[index].inputMode = FTM_SIGNAL_MEASUREMENT;
293  channelConfig[index].edgeAlignement = FTM_NO_PIN_CONTROL;
294  }
295 
296  /* Set channels configuration from user */
297  channelConfig[index].filterEn = configPtr->inputChConfig[index].filterEn;
298  channelConfig[index].filterValue = configPtr->inputChConfig[index].filterValue;
299  channelConfig[index].continuousModeEn = ((channel_extension_ftm_for_ic_t *)(configPtr->inputChConfig[index].channelExtension))->continuousModeEn;
300  channelConfig[index].channelsCallbacksParams = configPtr->inputChConfig[index].channelCallbackParams;
301  channelConfig[index].channelsCallbacks = configPtr->inputChConfig[index].channelCallbacks;
302 
303  if ((channelConfig[index].filterEn == true) && (channel < CHAN4_IDX))
304  {
305  /* Only sets the FTM filter clock is divided by 4 */
306  FTM_DRV_SetClockFilterPs(base, FTM_FILTER_PRESCALER_DIVIDE_BY_4);
307  }
308 
309  /* Store some needed information into state structure */
310  icState->channelConfigArray[index] = configPtr->inputChConfig[index].hwChannelId;
311  icState->icChannelType[channel] = configPtr->inputChConfig[index].inputCaptureMode;
312  icState->enableContinuousMode[channel] = channelConfig[index].continuousModeEn;
313  icState->enableNotification[channel] = true;
314  icState->channelCallbackParams[channel] = channelConfig[index].channelsCallbacksParams;
315  icState->channelCallbacks[channel] = channelConfig[index].channelsCallbacks;
316  }
317 
318  param.inputChConfig = channelConfig;
319  /* Configure channels in input capture mode */
320  status = FTM_DRV_InitInputCapture(instance,
321  &param);
322 
323  return status;
324 }

334 static status_t IC_EmiosInit(uint32_t instance,
335  const ic_config_t * configPtr)
336 {
337  DEV_ASSERT(instance < IC_PAL_INSTANCE_MAX);
338  ic_pal_state_t * icState;
339  ic_instance_t instancePtr;
340  status_t status = STATUS_SUCCESS;
341  uint8_t index = 0U;
342  static const IRQn_Type emiosIrq[eMIOS_INSTANCE_COUNT][(eMIOS_UC_COUNT / 2U)] = eMIOS_IRQS;
343 
344  /* Allocate one of the IC state structure for this instance */
345  instancePtr.instIdx = instance;
346  instancePtr.instType = IC_INST_TYPE_EMIOS;
347  index = FindIcState(&instancePtr);
348  icState = &g_icPalStatePtr[index];
349 
350  emios_common_param_t commonParam;
351  uint8_t channel = 0U;
352  uint8_t optionMode = 0U;
353  emios_bus_select_t timeBase = EMIOS_BUS_SEL_A;
354  emios_mc_mode_param_t mcParam;
355  emios_input_capture_param_t icParam;
356  emios_gpio_mode_param_t gpioParam;
357 
358  /* Get common parameters from user */
359  commonParam.allowDebugMode = false;
360  commonParam.lowPowerMode = false;
361  commonParam.clkDivVal = ((extension_emios_for_ic_t *)(configPtr->extension))->clkDivVal;
362  commonParam.enableGlobalPrescaler = ((extension_emios_for_ic_t *)(configPtr->extension))->enableGlobalPrescaler;
363  commonParam.enableGlobalTimeBase = ((extension_emios_for_ic_t *)(configPtr->extension))->enableGlobalTimeBase;
364 
365  /* Initialize the global for a eMIOS group */
366  EMIOS_DRV_InitGlobal((uint8_t)instance,
367  &commonParam);
368 
369  /* Get the information from user configuration */
370  mcParam.mode = EMIOS_MODE_MCB_UP_COUNTER_INT_CLK;
371  mcParam.period = MAX_COUNTER_VALUE;
372  mcParam.filterInput = EMIOS_INPUT_FILTER_BYPASS;
373  mcParam.filterEn = false;
374  mcParam.triggerMode = EMIOS_TRIGGER_EDGE_ANY;
375 
376  /* Configure a list of channels which are used */
377  for (index = 0U; index < configPtr->nNumChannels; index++)
378  {
379  mcParam.internalPrescaler = (emios_clock_internal_ps_t)((channel_extension_emios_for_ic_t *)(configPtr->inputChConfig[index].channelExtension))->prescaler;
380  mcParam.internalPrescalerEn = true;
381 
382  switch (((channel_extension_emios_for_ic_t *)(configPtr->inputChConfig[index].channelExtension))->timebase)
383  {
384  case IC_BUS_SEL_A:
385  /* Set channel to use as a time base */
386  channel = 23U;
387  timeBase = EMIOS_BUS_SEL_A;
388  break;
389  case IC_BUS_SEL_B:
390  /* Set channel to use as a time base */
391  channel = 0U;
392  timeBase = EMIOS_BUS_SEL_BCDE;
393  break;
394  case IC_BUS_SEL_C:
395  /* Set channel to use as a time base */
396  channel = 8U;
397  timeBase = EMIOS_BUS_SEL_BCDE;
398  break;
399  case IC_BUS_SEL_D:
400  /* Set channel to use as a time base */
401  channel = 16U;
402  timeBase = EMIOS_BUS_SEL_BCDE;
403  break;
404  case IC_BUS_SEL_E:
405  /* Set channel to use as a time base */
406  channel = 24U;
407  timeBase = EMIOS_BUS_SEL_BCDE;
408  break;
409  case IC_BUS_SEL_F:
410  /* Set channel to use as a time base */
411  channel = 22U;
412  timeBase = EMIOS_BUS_SEL_F;
413  break;
414  case IC_BUS_SEL_INTERNAL:
415  /* Set channel to use as a time base */
416  channel = configPtr->inputChConfig[index].hwChannelId;
417  timeBase = EMIOS_BUS_SEL_INTERNAL;
418  break;
419  default:
420  /* Do nothing */
421  break;
422  }
423 
424  /* Initialize the counter mode */
425  status = EMIOS_DRV_MC_InitCounterMode((uint8_t)instance,
426  channel,
427  &mcParam);
428 
429  channel = configPtr->inputChConfig[index].hwChannelId;
430  icParam.mode = EMIOS_MODE_IC;
431  icParam.timebase = timeBase;
432  icParam.filterInput = (emios_input_filter_t)(configPtr->inputChConfig[index].filterValue);
433  icParam.filterEn = configPtr->inputChConfig[index].filterEn;
434 
435  if (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_RISING_EDGE)
436  {
437  icParam.inputCaptureMode = EMIOS_CAPTURE_TRIGGER_EDGE_RISING;
438  }
439  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_FALLING_EDGE)
440  {
441  icParam.inputCaptureMode = EMIOS_CAPTURE_TRIGGER_EDGE_FALLING;
442  }
443  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_DISABLE_OPERATION)
444  {
445  /* Set default configure for an input pin */
446  gpioParam.mode = EMIOS_MODE_GPIO_INPUT;
447  gpioParam.filterEn = false;
448  gpioParam.filterInput = EMIOS_INPUT_FILTER_BYPASS;
449  gpioParam.triggerMode = EMIOS_TRIGGER_EDGE_ANY;
450 
451  /* Disable operation on the channel input */
452  EMIOS_DRV_InitGpioMode((uint8_t)instance,
453  channel,
454  &gpioParam);
455  }
456  else
457  {
458  optionMode = (uint8_t)configPtr->inputChConfig[index].inputCaptureMode - 1U;
459  icParam.inputCaptureMode = (emios_input_capture_mode_t)optionMode;
460  }
461 
462  if (configPtr->inputChConfig[index].inputCaptureMode != IC_DISABLE_OPERATION)
463  {
464  /* Initialize the input capture mode for each channel */
465  status = EMIOS_DRV_IC_InitInputCaptureMode((uint8_t)instance,
466  channel,
467  &icParam);
468  }
469 
470  EMIOS_DRV_ChannelEnableInterruptRequest((uint8_t)instance, channel);
471  /* Enable EMIOS interrupt */
472  INT_SYS_EnableIRQ(emiosIrq[instance][(channel>>1U)]);
473 
474  /* Store some needed information into state structure */
475  icState->channelConfigArray[index] = channel;
476  icState->icChannelType[channel] = configPtr->inputChConfig[index].inputCaptureMode;
477  icState->timeBaseSelection[channel] = icParam.timebase;
478  icState->filterEn[channel] = icParam.filterEn;
479  icState->filterInput[channel] = icParam.filterInput;
480  icState->enableNotification[channel] = true;
481  icState->channelCallbackParams[channel] = configPtr->inputChConfig[index].channelCallbackParams;
482  icState->channelCallbacks[channel] = configPtr->inputChConfig[index].channelCallbacks;
483  }
484 
485  /* Enable the global eMIOS to start counter */
486  EMIOS_DRV_EnableGlobalEmios((uint8_t)instance);
487 
488  return status;
489 }

565  userChannelConfig.interruptEnableMask = ETIMER_CH_IRQ_SOURCE_ICF1IE;

569  {

577  {
578  capCtr[0] = ETIMER_CPTMODE_BOTH;
579  }
580  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_MEASURE_RISING_EDGE_PERIOD)
581  {
582  capCtr[0] = ETIMER_CPTMODE_RISING;
583  userChannelConfig.captureWords = 2U;
584  }
585  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_MEASURE_FALLING_EDGE_PERIOD)
586  {
587  capCtr[0] = ETIMER_CPTMODE_FALLING;
588  userChannelConfig.captureWords = 2U;
589  }
590  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_MEASURE_PULSE_HIGH)
591  {
592  capCtr[0] = ETIMER_CPTMODE_RISING;
593  capCtr[1] = ETIMER_CPTMODE_FALLING;

595  }

597  {

608  {

615  }
